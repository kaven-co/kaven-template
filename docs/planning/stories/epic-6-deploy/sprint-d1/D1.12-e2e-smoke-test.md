## Technical Notes

**Script de smoke test completo:**
```bash
#!/bin/bash
# kaven-e2e-smoke-test.sh
set -euo pipefail

echo "Kaven E2E Smoke Test — $(date)"
echo "================================="

# Phase 1: CLI Bootstrap
echo "Phase 1: CLI Bootstrap"
npm install -g kaven-cli@latest
kaven doctor
kaven init kaven-smoke-test
cd kaven-smoke-test

# Verificar estrutura criada
test -f ".env" || (echo "FAIL: .env nao criado" && exit 1)
test -f "apps/api/package.json" || (echo "FAIL: apps/api/package.json nao encontrado" && exit 1)
test -f "packages/database/prisma/schema.prisma" || (echo "FAIL: schema.prisma nao encontrado" && exit 1)
echo "PASS: Phase 1 — CLI Bootstrap OK"

# Phase 2: Auth
echo "Phase 2: Authentication"
kaven auth login  # Interativo — abre browser
kaven auth whoami
echo "PASS: Phase 2 — Auth OK"

# Phase 3: Marketplace
echo "Phase 3: Marketplace"
kaven marketplace browse
kaven module install payments
test -f "packages/database/prisma/payments.prisma" || (echo "FAIL: payments.prisma nao mergeado" && exit 1)
echo "PASS: Phase 3 — Marketplace OK"

# Phase 4: App Runtime
echo "Phase 4: Application Runtime"
pnpm install
# Configurar .env com DATABASE_URL real para teste
pnpm db:migrate
pnpm dev &
DEV_PID=$!
sleep 10  # aguardar inicialização

curl -f http://localhost:3001 > /dev/null || (echo "FAIL: Tenant app nao iniciou" && kill $DEV_PID && exit 1)
curl -f http://localhost:3000 > /dev/null || (echo "FAIL: Admin app nao iniciou" && kill $DEV_PID && exit 1)
echo "PASS: Phase 4 — App Runtime OK"

# Phase 5: Manual check (interativo)
echo ""
echo "Phase 5: Manual User Flow Check"
echo "1. Abra http://localhost:3001/signup"
echo "2. Complete o signup com email real"
echo "3. Verifique se email chegou (< 2min)"
echo "4. Clique no link de verificacao"
echo "5. Faca login em http://localhost:3000 (admin)"
echo ""
read -p "Phase 5 passou? (y/n) " phase5_ok
[[ "$phase5_ok" == "y" ]] || (echo "FAIL: Phase 5 falhou" && kill $DEV_PID && exit 1)

# Phase 6: Production API
echo "Phase 6: Production API"
curl -f https://marketplace.kaven.site/health
curl -f https://marketplace.kaven.site/modules
echo "PASS: Phase 6 — Production API OK"

kill $DEV_PID 2>/dev/null
echo ""
echo "ALL PHASES PASSED — LAUNCH READY!"
echo "Deploy target: https://kaven.site"
```
