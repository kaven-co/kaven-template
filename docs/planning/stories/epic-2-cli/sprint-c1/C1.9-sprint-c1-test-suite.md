## Contexto

Sprint C1 replaces all mock implementations in kaven-cli with real marketplace API integration. Every new feature needs comprehensive test coverage to ensure reliability. This story creates a unified test suite covering auth flows, marketplace client, token management, install pipeline, license verification, lifecycle scripts, and env var injection.

Tests must mock external HTTP calls (marketplace API) but verify the CLI's internal logic thoroughly. The test approach uses MSW (Mock Service Worker) or nock for HTTP mocking, temp directories for filesystem tests, and fixture files for module packages.

## User Story

**As a** QA engineer,
**I want** comprehensive automated tests for all Sprint C1 features,
**So that** we can confidently ship the real marketplace integration without regressions.

### JWT Test Helper

```typescript
// tests/helpers/jwt.ts
export function createTestJwt(payload: Partial<JwtPayload>): string {
  const header = Buffer.from(JSON.stringify({ alg: 'RS256', typ: 'JWT' })).toString('base64url');
  const body = Buffer.from(JSON.stringify({
    sub: 'user_test',
    email: payload.email ?? 'test@kaven.sh',
    githubId: payload.githubId ?? 'testuser',
    tier: payload.tier ?? 'starter',
    iat: Math.floor(Date.now() / 1000),
    exp: payload.exp ?? Math.floor(Date.now() / 1000) + 86400, // 24h default
  })).toString('base64url');
  const signature = Buffer.from('test-signature').toString('base64url');
  return `${header}.${body}.${signature}`;
}

export function createExpiredJwt(): string {
  return createTestJwt({ exp: Math.floor(Date.now() / 1000) - 3600 }); // Expired 1h ago
}

export function createExpiringJwt(): string {
  return createTestJwt({ exp: Math.floor(Date.now() / 1000) + 120 }); // Expires in 2min
}
```

### Fixture Project Helper

```typescript
// tests/helpers/fixture-project.ts
import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';

export function createFixtureProject(): { dir: string; cleanup: () => void } {
  const dir = fs.mkdtempSync(path.join(os.tmpdir(), 'kaven-test-'));

  // Create anchor files that ModuleInstaller expects
  const anchors = {
    'package.json': '{ "name": "test-project", "dependencies": {} }',
    '.env': 'DATABASE_URL=postgresql://localhost:5432/test\n',
    'prisma/schema.prisma': `
      generator client { provider = "prisma-client-js" }
      datasource db { provider = "postgresql" url = env("DATABASE_URL") }
      // [KAVEN_MODULE_SCHEMA]
    `,
    'src/api/routes/index.ts': `
      // Routes
      // [KAVEN_MODULE:routes BEGIN]
      // [KAVEN_MODULE:routes END]
    `,
  };

  for (const [file, content] of Object.entries(anchors)) {
    const filePath = path.join(dir, file);
    fs.mkdirSync(path.dirname(filePath), { recursive: true });
    fs.writeFileSync(filePath, content);
  }

  return {
    dir,
    cleanup: () => fs.rmSync(dir, { recursive: true, force: true }),
  };
}
```

### Example Test Cases

```typescript
// tests/core/AuthService.test.ts
describe('AuthService', () => {
  let authService: AuthService;
  let tempDir: string;

  beforeEach(() => {
    tempDir = fs.mkdtempSync(path.join(os.tmpdir(), 'kaven-auth-test-'));
    authService = new AuthService(tempDir);
  });

  afterEach(() => fs.rmSync(tempDir, { recursive: true, force: true }));

  it('decodes JWT payload without signature verification', () => {
    const jwt = createTestJwt({ email: 'dev@kaven.sh', tier: 'pro' });
    authService.saveTokens({
      access_token: jwt,
      refresh_token: 'refresh',
      expires_at: new Date(Date.now() + 86400000).toISOString(),
      user: { email: 'dev@kaven.sh', githubId: 'devuser', tier: 'pro' },
    });
    const user = authService.getUserInfo();
    expect(user?.email).toBe('dev@kaven.sh');
    expect(user?.tier).toBe('pro');
  });

  it('returns token when not expiring soon', async () => {
    const jwt = createTestJwt({ exp: Math.floor(Date.now() / 1000) + 7200 });
    authService.saveTokens({
      access_token: jwt,
      refresh_token: 'refresh',
      expires_at: new Date(Date.now() + 7200000).toISOString(),
      user: { email: 'test@kaven.sh', githubId: 'test', tier: 'starter' },
    });
    const token = await authService.getValidToken();
    expect(token).toBe(jwt);
  });

  it('throws when not authenticated', async () => {
    await expect(authService.getValidToken()).rejects.toThrow('Not authenticated');
  });

  it('removes auth file on logout', () => {
    authService.saveTokens({ /* ... */ });
    authService.logout();
    expect(authService.isAuthenticated()).toBe(false);
  });
});
```

```typescript
// tests/infrastructure/MarketplaceClient.test.ts
describe('MarketplaceClient', () => {
  it('retries 3 times on 500 errors', async () => {
    let callCount = 0;
    mockServer.use(
      http.get('https://api.kaven.sh/modules', () => {
        callCount++;
        if (callCount < 4) return new HttpResponse(null, { status: 500 });
        return HttpResponse.json({ data: [], pagination: { page: 1, limit: 20, total: 0, totalPages: 0 } });
      })
    );

    const client = new MarketplaceClient(new AuthService());
    const result = await client.listModules();
    expect(callCount).toBe(4); // 1 original + 3 retries
    expect(result.data).toEqual([]);
  });

  it('does not retry on 401', async () => {
    mockServer.use(
      http.get('https://api.kaven.sh/modules/premium', () => {
        return HttpResponse.json({ message: 'Unauthorized' }, { status: 401 });
      })
    );

    const client = new MarketplaceClient(new AuthService());
    await expect(client.getModule('premium')).rejects.toThrow(AuthenticationError);
  });

  it('attaches auth header on authenticated requests', async () => {
    let capturedHeaders: Headers | null = null;
    mockServer.use(
      http.post('https://api.kaven.sh/download-tokens', ({ request }) => {
        capturedHeaders = request.headers;
        return HttpResponse.json({ token: 'dl-test', artifactUrl: '...', expiresAt: '...' });
      })
    );

    // Setup auth
    const authService = new AuthService(tempDir);
    authService.saveTokens({ access_token: 'test-jwt', /* ... */ });

    const client = new MarketplaceClient(authService);
    await client.createDownloadToken('mod1', 'rel1');

    expect(capturedHeaders?.get('authorization')).toBe('Bearer test-jwt');
  });
});
```
