# C1.7 — PostInstall/PreRemove Scripts

> **Epic:** 2 (CLI) | **Sprint:** C1 | **Agent:** @dev
> **Priority:** P1
> **Depends on:** None (standalone enhancement to module lifecycle)

---

## Contexto

Modules often need setup steps beyond just copying files — common examples include `pnpm install` to add new dependencies, `npx prisma generate` to regenerate the Prisma client after schema changes, and `npx prisma migrate dev` to apply database migrations. Currently, these steps are documented in module READMEs but users must run them manually. This story adds support for `postInstall` and `preRemove` lifecycle scripts defined in the module's `module.json` manifest, similar to npm lifecycle scripts.

Security is a concern: we run scripts with user confirmation, enforce timeouts, and capture output. Full sandboxing (network isolation, filesystem restrictions) is a future enhancement.

## User Story

**As a** module author,
**I want to** define lifecycle scripts in my module manifest,
**So that** required setup and teardown steps run automatically during install and removal.

---

## Acceptance Criteria

- [ ] `module.json` manifest supports `postInstall` and `preRemove` string arrays
- [ ] After file injection (`ModuleInstaller.install()`), `postInstall` scripts execute in order
- [ ] Before file removal (`ModuleInstaller.remove()`), `preRemove` scripts execute in order
- [ ] Scripts run in the project root directory (`process.cwd()`)
- [ ] Each script has a 60-second timeout; killed with SIGTERM on timeout
- [ ] stdout and stderr captured and streamed to user in real-time
- [ ] Before running scripts, user is prompted: `"Module 'payments' wants to run: [scripts]. Allow? [Y/n]"`
- [ ] `--yes` or `-y` flag skips confirmation (for CI environments)
- [ ] On script failure (non-zero exit): warn user but DO NOT rollback install
- [ ] Warning message: `"Script 'pnpm install' failed (exit code 1). You may need to run it manually."`
- [ ] Script output prefixed with `[postInstall]` or `[preRemove]` label
- [ ] Empty script arrays or missing fields: skip silently (no error)
- [ ] Scripts that don't exist in PATH: show clear error `"Command 'foo' not found"`

### Script Runner Implementation

```typescript
// src/core/ScriptRunner.ts
import { spawn } from 'child_process';
import chalk from 'chalk';

interface ScriptRunnerOptions {
  cwd: string;
  timeout: number;       // ms, default 60000
  skipConfirmation: boolean;
  label: 'postInstall' | 'preRemove';
}

export class ScriptRunner {
  private defaultTimeout = 60_000; // 60 seconds

  /**
   * Run an array of scripts sequentially with user confirmation.
   * Returns true if all scripts succeeded, false if any failed.
   */
  async runScripts(
    scripts: string[],
    options: ScriptRunnerOptions
  ): Promise<{ success: boolean; results: ScriptResult[] }> {
    if (!scripts || scripts.length === 0) {
      return { success: true, results: [] };
    }

    // Prompt user for confirmation
    if (!options.skipConfirmation) {
      console.log(
        chalk.yellow(`\n  Module wants to run ${options.label} scripts:`)
      );
      scripts.forEach((s, i) => console.log(chalk.dim(`    ${i + 1}. ${s}`)));
      console.log();

      const { confirm } = await import('@inquirer/prompts');
      const allowed = await confirm({
        message: 'Allow these scripts to run?',
        default: true,
      });

      if (!allowed) {
        console.log(chalk.dim('  Scripts skipped. You may need to run them manually.'));
        return {
          success: true,
          results: scripts.map(s => ({ script: s, status: 'skipped' as const })),
        };
      }
    }

    const results: ScriptResult[] = [];
    let allSuccess = true;

    for (const script of scripts) {
      const result = await this.runSingleScript(script, options);
      results.push(result);
      if (result.status === 'failed' || result.status === 'timeout') {
        allSuccess = false;
        // Continue with next script — don't abort
      }
    }

    return { success: allSuccess, results };
  }

  private async runSingleScript(
    script: string,
    options: ScriptRunnerOptions
  ): Promise<ScriptResult> {
    const label = chalk.dim(`[${options.label}]`);
    console.log(`${label} Running: ${chalk.bold(script)}`);

    return new Promise((resolve) => {
      const timeout = options.timeout ?? this.defaultTimeout;

      // Parse command and args
      const parts = script.split(/\s+/);
      const cmd = parts[0];
      const args = parts.slice(1);

      const child = spawn(cmd, args, {
        cwd: options.cwd,
        shell: true,
        stdio: ['ignore', 'pipe', 'pipe'],
        env: { ...process.env },
      });

      let stdout = '';
      let stderr = '';

      child.stdout.on('data', (data) => {
        const line = data.toString();
        stdout += line;
        process.stdout.write(chalk.dim(`${label} ${line}`));
      });

      child.stderr.on('data', (data) => {
        const line = data.toString();
        stderr += line;
        process.stderr.write(chalk.dim(`${label} ${line}`));
      });

      // Timeout handler
      const timer = setTimeout(() => {
        child.kill('SIGTERM');
        // Give 5s grace period, then SIGKILL
        setTimeout(() => {
          if (!child.killed) child.kill('SIGKILL');
        }, 5000);
      }, timeout);

      child.on('error', (error) => {
        clearTimeout(timer);
        if (error.message.includes('ENOENT')) {
          console.log(chalk.red(`${label} Command '${cmd}' not found`));
        } else {
          console.log(chalk.red(`${label} Error: ${error.message}`));
        }
        resolve({
          script,
          status: 'failed',
          exitCode: -1,
          error: error.message,
          stdout,
          stderr,
        });
      });

      child.on('close', (code, signal) => {
        clearTimeout(timer);

        if (signal === 'SIGTERM' || signal === 'SIGKILL') {
          console.log(
            chalk.yellow(`${label} Script '${script}' timed out after ${timeout / 1000}s`)
          );
          resolve({ script, status: 'timeout', exitCode: -1, stdout, stderr });
          return;
        }

        if (code === 0) {
          console.log(chalk.green(`${label} ✓ ${script}`));
          resolve({ script, status: 'success', exitCode: 0, stdout, stderr });
        } else {
          console.log(
            chalk.yellow(
              `${label} Script '${script}' failed (exit code ${code}). ` +
              'You may need to run it manually.'
            )
          );
          resolve({ script, status: 'failed', exitCode: code ?? 1, stdout, stderr });
        }
      });
    });
  }
}

interface ScriptResult {
  script: string;
  status: 'success' | 'failed' | 'timeout' | 'skipped';
  exitCode?: number;
  error?: string;
  stdout?: string;
  stderr?: string;
}
```

### Integration with ModuleInstaller

```typescript
// In ModuleInstaller.install() — after file injection
const scriptRunner = new ScriptRunner();
if (manifest.postInstall?.length) {
  const { success, results } = await scriptRunner.runScripts(manifest.postInstall, {
    cwd: projectDir,
    timeout: 60_000,
    skipConfirmation: options.yes ?? false,
    label: 'postInstall',
  });
  if (!success) {
    console.log(chalk.yellow('\n  Some post-install scripts failed. Module files are installed.'));
    console.log(chalk.dim('  Review the errors above and run failed scripts manually.\n'));
  }
}

// In ModuleInstaller.remove() — before file removal
if (manifest.preRemove?.length) {
  const { success } = await scriptRunner.runScripts(manifest.preRemove, {
    cwd: projectDir,
    timeout: 60_000,
    skipConfirmation: options.yes ?? false,
    label: 'preRemove',
  });
  // Proceed with removal regardless of script outcome
}
```
