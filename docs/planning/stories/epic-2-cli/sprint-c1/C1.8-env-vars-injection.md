# C1.8 — Environment Variables Injection

> **Epic:** 2 (CLI) | **Sprint:** C1 | **Agent:** @dev
> **Priority:** P2
> **Depends on:** None (standalone enhancement to module lifecycle)

---

## Contexto

Modules often require environment variables (API keys, database URLs, feature flags). Currently, module READMEs document required env vars, but users must manually add them to `.env` files. This story automates env var injection during module install/remove using the same marker-based approach that `MarkerService` uses for code injection. The `module.json` manifest already supports an `env` array — this story reads that array and manages `.env` files accordingly.

Sensitive values (API keys, secrets) are masked during input. The CLI never overwrites existing values without explicit confirmation.

## User Story

**As a** developer installing a Kaven module,
**I want** required environment variables to be automatically added to my `.env` files,
**So that** I don't have to manually look up and configure each variable.

---

## Acceptance Criteria

- [ ] Read `env` array from `module.json` manifest during install
- [ ] Each env entry has: `{ name, description, default?, sensitive?, required? }`
- [ ] For each env var: check if already present in `.env`
- [ ] If already present: skip silently (never overwrite without confirmation)
- [ ] If missing and has default: prompt user `"STRIPE_KEY (Stripe API key) [sk_test_...]: "`
- [ ] If missing and no default: prompt user `"STRIPE_KEY (Stripe API key): "`
- [ ] If `sensitive: true`: mask input with asterisks (password-style input)
- [ ] If `required: true` and user provides empty value: re-prompt
- [ ] Append new vars to `.env` with module marker comment:
  ```
  # [KAVEN_MODULE:payments BEGIN]
  STRIPE_KEY=sk_test_xxx
  STRIPE_WEBHOOK_SECRET=whsec_xxx
  # [KAVEN_MODULE:payments END]
  ```
- [ ] On module remove: remove env vars between module markers in `.env`
- [ ] Support multiple `.env` files: `.env`, `.env.local`, `.env.development`, `.env.production`
- [ ] User selects which `.env` file to write to (default: `.env`)
- [ ] `--env-file <path>` flag to specify target env file
- [ ] If `.env` file doesn't exist: create it
- [ ] Existing env vars with same name but different value: ask `"STRIPE_KEY already exists with a different value. Overwrite? [y/N]"`
- [ ] `--skip-env` flag skips env var injection entirely
- [ ] Summary after injection: `"Added 3 environment variables to .env"`

### EnvManager Implementation

```typescript
// src/core/EnvManager.ts
import * as fs from 'fs';
import * as path from 'path';
import chalk from 'chalk';

interface EnvVarDefinition {
  name: string;
  description: string;
  default?: string;
  sensitive?: boolean;
  required?: boolean;
}

interface EnvManagerOptions {
  projectDir: string;
  envFile?: string;       // default: '.env'
  skipEnv?: boolean;
  skipConfirmation?: boolean;
}

export class EnvManager {
  /**
   * Inject env vars for a module during install.
   */
  async injectEnvVars(
    moduleSlug: string,
    envVars: EnvVarDefinition[],
    options: EnvManagerOptions
  ): Promise<{ added: number; skipped: number }> {
    if (options.skipEnv || !envVars || envVars.length === 0) {
      return { added: 0, skipped: 0 };
    }

    const envFilePath = path.join(
      options.projectDir,
      options.envFile ?? '.env'
    );

    // Read existing env file (or empty)
    const existingContent = this.readEnvFile(envFilePath);
    const existingVars = this.parseEnvFile(existingContent);

    const { password, input, confirm } = await import('@inquirer/prompts');

    const newVars: Array<{ name: string; value: string }> = [];
    let skipped = 0;

    console.log(chalk.bold(`\n  Environment variables for '${moduleSlug}':\n`));

    for (const envDef of envVars) {
      // Check if already exists
      if (existingVars.has(envDef.name)) {
        const existingValue = existingVars.get(envDef.name)!;
        console.log(chalk.dim(`  ${envDef.name} — already set, skipping`));
        skipped++;
        continue;
      }

      // Prompt for value
      const promptMessage = envDef.default
        ? `${envDef.name} (${envDef.description}) [${envDef.sensitive ? '****' : envDef.default}]`
        : `${envDef.name} (${envDef.description})`;

      let value: string;

      if (envDef.sensitive) {
        value = await password({
          message: promptMessage,
        });
        if (!value && envDef.default) value = envDef.default;
      } else {
        value = await input({
          message: promptMessage,
          default: envDef.default,
        });
      }

      // Validate required
      if (envDef.required && !value) {
        // Re-prompt once
        console.log(chalk.yellow(`  ${envDef.name} is required.`));
        value = envDef.sensitive
          ? await password({ message: `${envDef.name}: ` })
          : await input({ message: `${envDef.name}: ` });

        if (!value) {
          console.log(chalk.yellow(`  Skipping ${envDef.name} — you'll need to set it manually.`));
          skipped++;
          continue;
        }
      }

      newVars.push({ name: envDef.name, value });
    }

    if (newVars.length === 0) {
      console.log(chalk.dim('  No new environment variables to add.'));
      return { added: 0, skipped };
    }

    // Build marker block
    const markerBlock = this.buildMarkerBlock(moduleSlug, newVars);

    // Append to env file
    this.appendToEnvFile(envFilePath, existingContent, markerBlock);

    console.log(
      chalk.green(`\n  Added ${newVars.length} environment variable(s) to ${options.envFile ?? '.env'}`)
    );

    return { added: newVars.length, skipped };
  }

  /**
   * Remove env vars for a module during uninstall.
   */
  removeEnvVars(moduleSlug: string, options: EnvManagerOptions): number {
    const envFiles = ['.env', '.env.local', '.env.development', '.env.production'];
    let totalRemoved = 0;

    for (const envFile of envFiles) {
      const envFilePath = path.join(options.projectDir, envFile);
      if (!fs.existsSync(envFilePath)) continue;

      const content = fs.readFileSync(envFilePath, 'utf-8');
      const beginMarker = `# [KAVEN_MODULE:${moduleSlug} BEGIN]`;
      const endMarker = `# [KAVEN_MODULE:${moduleSlug} END]`;

      const beginIdx = content.indexOf(beginMarker);
      const endIdx = content.indexOf(endMarker);

      if (beginIdx === -1 || endIdx === -1) continue;

      // Count vars being removed
      const block = content.substring(beginIdx, endIdx + endMarker.length);
      const varCount = block.split('\n').filter(l => l.match(/^[A-Z_]+=/) ).length;

      // Remove block (including surrounding newlines)
      const before = content.substring(0, beginIdx).replace(/\n+$/, '\n');
      const after = content.substring(endIdx + endMarker.length + 1);
      const newContent = before + after;

      fs.writeFileSync(envFilePath, newContent);
      totalRemoved += varCount;

      if (varCount > 0) {
        console.log(chalk.dim(`  Removed ${varCount} env var(s) from ${envFile}`));
      }
    }

    return totalRemoved;
  }

  private readEnvFile(filePath: string): string {
    try {
      return fs.readFileSync(filePath, 'utf-8');
    } catch {
      return '';
    }
  }

  private parseEnvFile(content: string): Map<string, string> {
    const vars = new Map<string, string>();
    for (const line of content.split('\n')) {
      const match = line.match(/^([A-Z_][A-Z0-9_]*)=(.*)$/);
      if (match) {
        vars.set(match[1], match[2]);
      }
    }
    return vars;
  }

  private buildMarkerBlock(
    moduleSlug: string,
    vars: Array<{ name: string; value: string }>
  ): string {
    const lines = [
      `# [KAVEN_MODULE:${moduleSlug} BEGIN]`,
      ...vars.map(v => `${v.name}=${v.value}`),
      `# [KAVEN_MODULE:${moduleSlug} END]`,
    ];
    return lines.join('\n');
  }

  private appendToEnvFile(filePath: string, existingContent: string, block: string): void {
    const dir = path.dirname(filePath);
    fs.mkdirSync(dir, { recursive: true });

    const separator = existingContent.endsWith('\n') || existingContent === '' ? '\n' : '\n\n';
    const newContent = existingContent + separator + block + '\n';
    fs.writeFileSync(filePath, newContent);
  }
}
```
