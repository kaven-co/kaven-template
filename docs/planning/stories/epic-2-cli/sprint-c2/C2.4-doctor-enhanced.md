# C2.4 — `kaven doctor` Enhanced

> **Epic:** 2 (CLI) | **Sprint:** C2 | **Agent:** @dev
> **Priority:** P1
> **Depends on:** C1.6 (License Verification)

---

## Contexto

The current `kaven doctor` command performs basic health checks: verifying anchor files exist, checking marker integrity, and validating that dependencies are installed. With the new marketplace integration (Sprint C1), there are additional dimensions to check: schema merge integrity, environment variable completeness, license validity, framework version compatibility, and Prisma client synchronization.

The `--fix` flag exists but is mostly a no-op. This story makes it functional for auto-fixable issues while providing a clear, colorized report for all checks.

## User Story

**As a** developer maintaining a Kaven project,
**I want** the `doctor` command to perform comprehensive health checks and auto-fix common issues,
**So that** I can quickly diagnose and resolve configuration problems.

---

## Acceptance Criteria

- [ ] All existing checks preserved: anchors, markers, dependencies
- [ ] New check — Schema merge integrity:
  - Verify base Prisma schema is valid
  - Verify each installed module's schema extension is present
  - Detect duplicate model definitions across modules
  - Detect missing relations or foreign key references
- [ ] New check — Env var completeness:
  - Read all installed modules' `module.json` env requirements
  - Verify each required env var exists in `.env`
  - Warn for optional but missing env vars
  - Flag env vars with placeholder/default values that should be customized
- [ ] New check — License validity:
  - Check if authenticated
  - Verify license not expired
  - Warn if license expires within 30 days
  - Check if installed modules match license tier
- [ ] New check — Framework version compatibility:
  - Read each module's `kavenVersion` from manifest
  - Compare against current framework version in `package.json`
  - Flag incompatible modules (semver mismatch)
- [ ] New check — Prisma client sync:
  - Detect if `prisma/schema.prisma` was modified after last `prisma generate`
  - Compare schema file mtime vs generated client mtime
  - Flag if Prisma client is stale
- [ ] `--fix` flag auto-fixes:
  - Missing dependencies → runs `pnpm install`
  - Stale Prisma client → runs `npx prisma generate`
  - Missing `.env` vars → prompts for values (delegating to EnvManager)
- [ ] Issues that `--fix` cannot resolve show manual instructions
- [ ] Output format: colorized report with severity levels:
  - `[ERROR]` (red) — Must fix before app will work
  - `[WARN]` (yellow) — Should fix, may cause issues
  - `[INFO]` (dim) — Informational, no action needed
  - `[OK]` (green) — Check passed
- [ ] Exit codes:
  - `0` — All checks pass (only OK and INFO)
  - `1` — Errors found
  - `2` — Warnings found (no errors)
- [ ] Summary at end: `"5 checks passed, 1 warning, 0 errors"`
- [ ] `--json` flag outputs machine-readable JSON report

## Technical Notes

### Check Registry Architecture

```typescript
// src/core/doctor/CheckRegistry.ts

interface CheckResult {
  name: string;
  status: 'ok' | 'warn' | 'error' | 'info';
  message: string;
  fixable: boolean;
  fixCommand?: string;
  details?: string[];
}

interface HealthCheck {
  name: string;
  description: string;
  run(context: DoctorContext): Promise<CheckResult>;
  fix?(context: DoctorContext): Promise<boolean>;
}

interface DoctorContext {
  projectDir: string;
  installedModules: ModuleManifest[];
  envVars: Map<string, string>;
  frameworkVersion: string;
  authService: AuthService;
  licenseService: LicenseService;
}

export class CheckRegistry {
  private checks: HealthCheck[] = [];

  register(check: HealthCheck): void {
    this.checks.push(check);
  }

  async runAll(context: DoctorContext, fix: boolean): Promise<CheckResult[]> {
    const results: CheckResult[] = [];

    for (const check of this.checks) {
      const result = await check.run(context);
      results.push(result);

      if (fix && result.status !== 'ok' && result.fixable && check.fix) {
        console.log(chalk.dim(`  Attempting fix for: ${check.name}...`));
        const fixed = await check.fix(context);
        if (fixed) {
          result.status = 'ok';
          result.message += ' (fixed)';
        }
      }
    }

    return results;
  }
}
```

```typescript
// src/core/doctor/checks/schema-merge.ts
export const schemaMergeCheck: HealthCheck = {
  name: 'Schema Merge Integrity',
  description: 'Verify Prisma schema is valid with all module extensions',

  async run(context): Promise<CheckResult> {
    const schemaPath = path.join(context.projectDir, 'prisma', 'schema.prisma');
    if (!fs.existsSync(schemaPath)) {
      return { name: this.name, status: 'error', message: 'prisma/schema.prisma not found', fixable: false };
    }

    const schema = fs.readFileSync(schemaPath, 'utf-8');
    const issues: string[] = [];

    // Check each installed module has its schema section
    for (const mod of context.installedModules) {
      const marker = `// [KAVEN_MODULE:${mod.slug}`;
      if (!schema.includes(marker)) {
        issues.push(`Module '${mod.slug}' schema section missing`);
      }
    }

    // Check for duplicate model names
    const modelNames = [...schema.matchAll(/^model\s+(\w+)\s*\{/gm)].map(m => m[1]);
    const duplicates = modelNames.filter((n, i) => modelNames.indexOf(n) !== i);
    if (duplicates.length > 0) {
      issues.push(`Duplicate models: ${duplicates.join(', ')}`);
    }

    if (issues.length > 0) {
      return {
        name: this.name,
        status: 'error',
        message: `${issues.length} schema issue(s) found`,
        fixable: false,
        details: issues,
      };
    }

    return { name: this.name, status: 'ok', message: 'Schema is valid', fixable: false };
  },
};

// src/core/doctor/checks/env-completeness.ts
export const envCompletenessCheck: HealthCheck = {
  name: 'Environment Variables',
  description: 'Check all required env vars are present',

  async run(context): Promise<CheckResult> {
    const missing: string[] = [];
    const placeholder: string[] = [];

    for (const mod of context.installedModules) {
      const envDefs = mod.env ?? [];
      for (const envDef of envDefs) {
        const value = context.envVars.get(envDef.name);
        if (!value && envDef.required) {
          missing.push(`${envDef.name} (${mod.slug})`);
        } else if (value && envDef.default && value === envDef.default) {
          placeholder.push(`${envDef.name} still has default value`);
        }
      }
    }

    if (missing.length > 0) {
      return {
        name: this.name,
        status: 'error',
        message: `${missing.length} required env var(s) missing`,
        fixable: true,
        fixCommand: 'Prompt for missing values',
        details: missing,
      };
    }

    if (placeholder.length > 0) {
      return {
        name: this.name,
        status: 'warn',
        message: `${placeholder.length} env var(s) still have default values`,
        fixable: false,
        details: placeholder,
      };
    }

    return { name: this.name, status: 'ok', message: 'All env vars present', fixable: false };
  },

  async fix(context): Promise<boolean> {
    const envManager = new EnvManager();
    // Collect all missing required env vars and prompt
    for (const mod of context.installedModules) {
      const missingDefs = (mod.env ?? []).filter(
        e => e.required && !context.envVars.has(e.name)
      );
      if (missingDefs.length > 0) {
        await envManager.injectEnvVars(mod.slug, missingDefs, {
          projectDir: context.projectDir,
        });
      }
    }
    return true;
  },
};

// src/core/doctor/checks/prisma-sync.ts
export const prismaSyncCheck: HealthCheck = {
  name: 'Prisma Client Sync',
  description: 'Check if Prisma client matches current schema',

  async run(context): Promise<CheckResult> {
    const schemaPath = path.join(context.projectDir, 'prisma', 'schema.prisma');
    const clientPath = path.join(context.projectDir, 'node_modules', '.prisma', 'client');

    if (!fs.existsSync(clientPath)) {
      return {
        name: this.name,
        status: 'error',
        message: 'Prisma client not generated',
        fixable: true,
        fixCommand: 'npx prisma generate',
      };
    }

    const schemaMtime = fs.statSync(schemaPath).mtimeMs;
    const clientMtime = fs.statSync(clientPath).mtimeMs;

    if (schemaMtime > clientMtime) {
      return {
        name: this.name,
        status: 'warn',
        message: 'Prisma client is stale (schema modified after last generate)',
        fixable: true,
        fixCommand: 'npx prisma generate',
      };
    }

    return { name: this.name, status: 'ok', message: 'Prisma client is up to date', fixable: false };
  },

  async fix(context): Promise<boolean> {
    try {
      execSync('npx prisma generate', { cwd: context.projectDir, stdio: 'pipe' });
      return true;
    } catch {
      return false;
    }
  },
};
```
