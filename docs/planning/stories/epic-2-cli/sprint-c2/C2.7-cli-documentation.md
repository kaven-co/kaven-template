# C2.7 — CLI Documentation

> **Epic:** 2 (CLI) | **Sprint:** C2 | **Agent:** @dev
> **Priority:** P1
> **Depends on:** C2.1-C2.6 (all Sprint C2 feature stories)

---

## Contexto

With Sprints C1 and C2, kaven-cli grows from a prototype with mocked commands to a full-featured CLI tool. Every command needs comprehensive `--help` text, consistent examples, and clear error messages. The README needs a complete overhaul to serve as both a quick-start guide and a command reference. Error messages should guide users toward the right command rather than leaving them stuck.

Good CLI documentation is a multiplier — it reduces support burden and increases adoption. This story treats `--help` text as a first-class product surface.

## User Story

**As a** developer using kaven-cli for the first time,
**I want** every command to have clear help text with examples,
**So that** I can learn the tool without leaving the terminal.

// src/commands/marketplace/index.ts
import { Command } from 'commander';

export function registerMarketplaceCommands(program: Command) {
  const marketplace = program
    .command('marketplace')
    .alias('mp')
    .description('Browse and install modules from the Kaven marketplace');

  marketplace
    .command('list')
    .description('List available modules')
    .option('-c, --category <category>', 'Filter by category')
    .option('-s, --sort <sort>', 'Sort: newest, popular, name', 'newest')
    .option('-p, --page <page>', 'Page number', '1')
    .option('-l, --limit <limit>', 'Results per page (max 100)', '20')
    .option('--json', 'Output raw JSON for scripting')
    .option('--offline', 'Use cached data only')
    .option('--refresh', 'Force fresh data (ignore cache)')
    .addHelpText('after', `
Examples:
  $ kaven marketplace list
  $ kaven marketplace list --category payments --sort popular
  $ kaven marketplace list --json | jq '.data[].slug'
  $ kaven marketplace list --offline
    `)
    .action(listCommand);

  marketplace
    .command('install <slug>')
    .description('Install a module from the marketplace')
    .option('-v, --version <version>', 'Specific version to install')
    .option('-f, --force', 'Overwrite existing module without confirmation')
    .option('--skip-env', 'Skip environment variable injection')
    .option('--env-file <path>', 'Target env file', '.env')
    .option('-y, --yes', 'Auto-confirm all prompts')
    .addHelpText('after', `
Examples:
  $ kaven marketplace install payments
  $ kaven marketplace install analytics --version 2.0.0
  $ kaven marketplace install notifications --force --yes
    `)
    .action(installCommand);

  marketplace
    .command('browse')
    .description('Interactive module browser (TUI)')
    .addHelpText('after', `
Examples:
  $ kaven marketplace browse
    `)
    .action(browseCommand);
}

### Error Message Helpers

```typescript
// src/utils/error-messages.ts
import chalk from 'chalk';

export function formatError(message: string, suggestion?: string): string {
  let output = chalk.red(`  Error: ${message}`);
  if (suggestion) {
    output += `\n  ${chalk.dim(`Try: ${suggestion}`)}`;
  }
  return output;
}

export function suggestCommand(input: string, validCommands: string[]): string | null {
  // Levenshtein distance for "did you mean?" suggestions
  let minDist = Infinity;
  let closest = '';

  for (const cmd of validCommands) {
    const dist = levenshtein(input.toLowerCase(), cmd.toLowerCase());
    if (dist < minDist && dist <= 3) {
      minDist = dist;
      closest = cmd;
    }
  }

  return closest || null;
}

function levenshtein(a: string, b: string): number {
  const matrix: number[][] = [];
  for (let i = 0; i <= a.length; i++) matrix[i] = [i];
  for (let j = 0; j <= b.length; j++) matrix[0][j] = j;

  for (let i = 1; i <= a.length; i++) {
    for (let j = 1; j <= b.length; j++) {
      matrix[i][j] = Math.min(
        matrix[i - 1][j] + 1,
        matrix[i][j - 1] + 1,
        matrix[i - 1][j - 1] + (a[i - 1] === b[j - 1] ? 0 : 1)
      );
    }
  }

  return matrix[a.length][b.length];
}

// Usage in main command handler
program.on('command:*', (operands) => {
  const unknown = operands[0];
  const suggestion = suggestCommand(unknown, program.commands.map(c => c.name()));
  console.error(formatError(
    `Unknown command '${unknown}'.`,
    suggestion
      ? `Did you mean 'kaven ${suggestion}'? Run 'kaven --help' for all commands.`
      : `Run 'kaven --help' to see available commands.`
  ));
  process.exit(1);
});
```
