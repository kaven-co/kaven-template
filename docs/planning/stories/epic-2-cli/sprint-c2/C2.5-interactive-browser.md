# C2.5 — Interactive Module Browser

> **Epic:** 2 (CLI) | **Sprint:** C2 | **Agent:** @dev
> **Priority:** P2
> **Depends on:** C1.4 (Marketplace List Real)

---

## Contexto

While `kaven marketplace list` provides a static table view, developers often want to explore modules interactively — browsing through categories, reading descriptions, and installing directly from the browser interface. The `kaven marketplace browse` command provides a TUI (Terminal User Interface) experience using interactive prompts, allowing keyboard-driven navigation through the module catalog.

This is a UX polish feature that enhances discoverability. It uses the same `MarketplaceClient` API from C1.2 but presents data in an interactive, filterable format.

## User Story

**As a** developer exploring the Kaven marketplace,
**I want** an interactive terminal browser for modules,
**So that** I can discover, compare, and install modules without switching to a web browser.

---

## Acceptance Criteria

- [ ] `kaven marketplace browse` launches interactive TUI
- [ ] Initial view: category selection (or "All modules")
- [ ] Categories fetched from API: `GET /modules/categories`
- [ ] Category list with arrow key navigation and Enter to select
- [ ] After category: show paginated module list within that category
- [ ] Module list shows: name, version, tier badge, short description (truncated to terminal width)
- [ ] Arrow keys to navigate, Enter to select a module for details
- [ ] Module detail view shows:
  - Full name and slug
  - Version and last updated date
  - Author name
  - Full description
  - Tier requirement (with color badge)
  - Install count
  - Dependencies list
  - Env vars required
- [ ] Action menu on detail view:
  - "Install this module" → delegates to `installCommand(slug)`
  - "Back to list" → return to module list
  - "Exit" → close browser
- [ ] Search: typing `/` activates search mode, filters module list by name/description
- [ ] Search is client-side filtering on currently loaded page
- [ ] Pagination: "Load more" option at bottom of list (fetches next page)
- [ ] Empty category: show `"No modules in this category yet."`
- [ ] Network error: show error and offer retry
- [ ] `Esc` key returns to previous level (detail → list → categories → exit)
- [ ] `q` key exits at any level

## Technical Notes

### Implementation with @inquirer/prompts

```typescript
// src/commands/marketplace/browse.ts
import { select, search } from '@inquirer/prompts';
import chalk from 'chalk';
import { MarketplaceClient } from '../../infrastructure/MarketplaceClient';
import { AuthService } from '../../core/AuthService';
import { installCommand } from './install';

export async function browseCommand() {
  const authService = new AuthService();
  const client = new MarketplaceClient(authService);

  console.log(chalk.bold('\n  Kaven Module Browser\n'));

  while (true) {
    // Level 1: Category selection
    const category = await selectCategory(client);
    if (category === '__exit__') break;

    // Level 2: Module list within category
    while (true) {
      const moduleSlug = await selectModule(client, category);
      if (moduleSlug === '__back__') break;
      if (moduleSlug === '__exit__') return;

      // Level 3: Module detail
      const action = await showModuleDetail(client, moduleSlug);
      if (action === 'install') {
        await installCommand(moduleSlug, {});
        // After install, return to list
      }
      if (action === 'exit') return;
      // 'back' continues the inner loop
    }
  }
}

async function selectCategory(client: MarketplaceClient): Promise<string> {
  try {
    const categories = await client.getCategories();

    const choices = [
      { value: '__all__', name: chalk.bold('All modules') },
      ...categories.map(cat => ({
        value: cat.slug,
        name: `${cat.name} ${chalk.dim(`(${cat.count} modules)`)}`,
      })),
      { value: '__exit__', name: chalk.dim('Exit') },
    ];

    return await select({
      message: 'Select a category:',
      choices,
      pageSize: 15,
    });
  } catch {
    console.error(chalk.red('  Could not load categories. Check your connection.'));
    return '__exit__';
  }
}

async function selectModule(
  client: MarketplaceClient,
  category: string
): Promise<string> {
  const filter = category === '__all__' ? undefined : category;

  try {
    const result = await client.listModules({
      category: filter,
      limit: 20,
      sort: 'popular',
    });

    if (result.data.length === 0) {
      console.log(chalk.dim('\n  No modules in this category yet.\n'));
      return '__back__';
    }

    const choices = [
      ...result.data.map(mod => ({
        value: mod.slug,
        name: formatModuleChoice(mod),
        description: mod.description?.substring(0, 60),
      })),
      { value: '__back__', name: chalk.dim('← Back to categories') },
      { value: '__exit__', name: chalk.dim('Exit') },
    ];

    return await select({
      message: `Modules${filter ? ` in ${filter}` : ''}:`,
      choices,
      pageSize: 12,
    });
  } catch {
    console.error(chalk.red('  Could not load modules.'));
    return '__back__';
  }
}

function formatModuleChoice(mod: Module): string {
  const tierBadge = {
    starter: chalk.green('[S]'),
    complete: chalk.yellow('[C]'),
    pro: chalk.magenta('[P]'),
  }[mod.tier] ?? '';

  const installs = mod.installs >= 1000
    ? `${(mod.installs / 1000).toFixed(1)}k`
    : String(mod.installs);

  return [
    mod.name.padEnd(28),
    chalk.dim(`v${mod.version}`.padEnd(8)),
    tierBadge.padEnd(5),
    chalk.dim(`${installs} installs`),
  ].join('  ');
}

async function showModuleDetail(
  client: MarketplaceClient,
  slug: string
): Promise<'install' | 'back' | 'exit'> {
  try {
    const mod = await client.getModule(slug);

    console.log(chalk.bold(`\n  ${mod.name}`));
    console.log(chalk.dim(`  ${mod.slug} v${mod.version}\n`));
    console.log(`  ${mod.description}\n`);
    console.log(`  Author:    ${mod.author?.name ?? 'Unknown'}`);
    console.log(`  Tier:      ${colorTier(mod.tier)}`);
    console.log(`  Installs:  ${mod.installs}`);
    console.log(`  Category:  ${mod.category}`);

    if (mod.dependencies && Object.keys(mod.dependencies).length > 0) {
      console.log(`  Deps:      ${Object.keys(mod.dependencies).join(', ')}`);
    }

    if (mod.env && mod.env.length > 0) {
      console.log(`  Env vars:  ${mod.env.length} required`);
    }

    console.log();

    return await select({
      message: 'Action:',
      choices: [
        { value: 'install' as const, name: chalk.green('Install this module') },
        { value: 'back' as const, name: 'Back to list' },
        { value: 'exit' as const, name: chalk.dim('Exit') },
      ],
    });
  } catch {
    console.error(chalk.red(`  Could not load details for '${slug}'.`));
    return 'back';
  }
}

function colorTier(tier: string): string {
  switch (tier) {
    case 'starter': return chalk.green('Starter');
    case 'complete': return chalk.yellow('Complete');
    case 'pro': return chalk.magenta('Pro');
    default: return tier;
  }
}
```
