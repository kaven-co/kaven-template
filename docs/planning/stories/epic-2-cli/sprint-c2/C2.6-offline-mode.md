# C2.6 — Offline Mode

> **Epic:** 2 (CLI) | **Sprint:** C2 | **Agent:** @dev
> **Priority:** P2
> **Depends on:** C1.2 (Real MarketplaceClient)

---

## Contexto

Developers don't always have reliable internet access — on flights, in areas with spotty connectivity, or when marketplace API is temporarily down. The CLI should gracefully degrade by caching API responses and serving cached data when the network is unavailable. This is not about offline installation (which requires the actual artifact), but about allowing browsing and planning (list, search, details) to work with cached data.

The cache uses a simple JSON file system with configurable TTL and LRU eviction to prevent unbounded cache growth.

## User Story

**As a** developer with intermittent connectivity,
**I want** the CLI to serve cached marketplace data when offline,
**So that** I can browse modules and plan installations even without internet access.

---

## Acceptance Criteria

- [ ] Module listing cached in `~/.kaven/cache/modules.json` after successful API call
- [ ] Module manifests cached in `~/.kaven/cache/manifests/{slug}/{version}.json`
- [ ] Module details cached in `~/.kaven/cache/modules/{slug}.json`
- [ ] Cache TTL: 24 hours for module listings, 7 days for manifests (versioned, immutable)
- [ ] On network error during API call: automatically try cache (transparent fallback)
- [ ] When serving cached data, show warning: `"Using cached data from 2h ago"`
- [ ] `kaven marketplace list --offline` forces cached data (no API call attempted)
- [ ] `kaven marketplace list --refresh` forces fresh data (ignores cache)
- [ ] Cache is updated on every successful API call (write-through)
- [ ] Cache size limit: 50 MB total
- [ ] LRU eviction: when cache exceeds limit, remove least-recently-accessed entries
- [ ] Cache metadata stored in `~/.kaven/cache/meta.json`:
  - Total size in bytes
  - Entry count
  - Last cleanup timestamp
- [ ] `kaven cache status` shows cache size, entry count, oldest entry
- [ ] `kaven cache clear` deletes all cached data
- [ ] Cache directory permissions: `0700` (owner only)

### CacheManager Implementation

```typescript
// src/core/CacheManager.ts
import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';

interface CacheEntry<T> {
  data: T;
  cachedAt: string;      // ISO 8601
  expiresAt: string;     // ISO 8601
  accessedAt: string;    // ISO 8601 — for LRU
  sizeBytes: number;
  key: string;
}

interface CacheMeta {
  totalSizeBytes: number;
  entryCount: number;
  lastCleanup: string;
}

const DEFAULT_TTL_MS = {
  listing: 24 * 60 * 60 * 1000,      // 24 hours
  detail: 24 * 60 * 60 * 1000,       // 24 hours
  manifest: 7 * 24 * 60 * 60 * 1000, // 7 days (versioned = immutable)
};

const MAX_CACHE_SIZE_BYTES = 50 * 1024 * 1024; // 50 MB

export class CacheManager {
  private cacheDir: string;
  private metaPath: string;

  constructor(baseDir = path.join(os.homedir(), '.kaven', 'cache')) {
    this.cacheDir = baseDir;
    this.metaPath = path.join(baseDir, 'meta.json');
  }

  /**
   * Get cached data if fresh. Returns null if expired or missing.
   */
  get<T>(key: string): { data: T; age: string } | null {
    const entryPath = this.keyToPath(key);
    try {
      const raw = fs.readFileSync(entryPath, 'utf-8');
      const entry: CacheEntry<T> = JSON.parse(raw);

      // Check expiry
      if (new Date(entry.expiresAt).getTime() < Date.now()) {
        return null; // Expired
      }

      // Update access time for LRU
      entry.accessedAt = new Date().toISOString();
      fs.writeFileSync(entryPath, JSON.stringify(entry));

      // Calculate human-readable age
      const ageMs = Date.now() - new Date(entry.cachedAt).getTime();
      const age = this.formatAge(ageMs);

      return { data: entry.data, age };
    } catch {
      return null;
    }
  }

  /**
   * Get cached data even if expired (for offline fallback).
   */
  getStale<T>(key: string): { data: T; age: string } | null {
    const entryPath = this.keyToPath(key);
    try {
      const raw = fs.readFileSync(entryPath, 'utf-8');
      const entry: CacheEntry<T> = JSON.parse(raw);
      const ageMs = Date.now() - new Date(entry.cachedAt).getTime();
      return { data: entry.data, age: this.formatAge(ageMs) };
    } catch {
      return null;
    }
  }

  /**
   * Store data in cache.
   */
  set<T>(key: string, data: T, ttlMs?: number): void {
    const entryPath = this.keyToPath(key);
    const dir = path.dirname(entryPath);
    fs.mkdirSync(dir, { recursive: true, mode: 0o700 });

    const serialized = JSON.stringify(data);
    const entry: CacheEntry<T> = {
      data,
      key,
      cachedAt: new Date().toISOString(),
      expiresAt: new Date(Date.now() + (ttlMs ?? DEFAULT_TTL_MS.listing)).toISOString(),
      accessedAt: new Date().toISOString(),
      sizeBytes: Buffer.byteLength(serialized),
    };

    fs.writeFileSync(entryPath, JSON.stringify(entry));

    // Check size and evict if needed
    this.evictIfNeeded();
  }

  /**
   * Clear all cached data.
   */
  clear(): void {
    try {
      fs.rmSync(this.cacheDir, { recursive: true, force: true });
      console.log('Cache cleared.');
    } catch {
      // Already clean
    }
  }

  /**
   * Get cache statistics.
   */
  getStatus(): { size: string; entries: number; oldestEntry: string | null } {
    let totalSize = 0;
    let entryCount = 0;
    let oldestTime = Date.now();

    this.walkDir(this.cacheDir, (filePath) => {
      try {
        const stats = fs.statSync(filePath);
        totalSize += stats.size;
        entryCount++;
        if (stats.mtimeMs < oldestTime) oldestTime = stats.mtimeMs;
      } catch { /* skip */ }
    });

    return {
      size: this.formatSize(totalSize),
      entries: entryCount,
      oldestEntry: entryCount > 0 ? this.formatAge(Date.now() - oldestTime) : null,
    };
  }

  private evictIfNeeded(): void {
    let totalSize = 0;
    const entries: Array<{ path: string; accessedAt: number; size: number }> = [];

    this.walkDir(this.cacheDir, (filePath) => {
      if (filePath === this.metaPath) return;
      try {
        const stats = fs.statSync(filePath);
        const raw = fs.readFileSync(filePath, 'utf-8');
        const entry = JSON.parse(raw);
        totalSize += stats.size;
        entries.push({
          path: filePath,
          accessedAt: new Date(entry.accessedAt ?? entry.cachedAt).getTime(),
          size: stats.size,
        });
      } catch { /* skip corrupt entries */ }
    });

    if (totalSize <= MAX_CACHE_SIZE_BYTES) return;

    // Sort by accessedAt ascending (oldest access first = evict first)
    entries.sort((a, b) => a.accessedAt - b.accessedAt);

    // Evict until under limit
    let currentSize = totalSize;
    for (const entry of entries) {
      if (currentSize <= MAX_CACHE_SIZE_BYTES * 0.8) break; // Evict to 80%
      try {
        fs.unlinkSync(entry.path);
        currentSize -= entry.size;
      } catch { /* skip */ }
    }
  }

  private keyToPath(key: string): string {
    // e.g. "modules:list:page1" → "modules/list/page1.json"
    const parts = key.split(':');
    return path.join(this.cacheDir, ...parts) + '.json';
  }

  private formatAge(ms: number): string {
    const minutes = Math.floor(ms / 60000);
    if (minutes < 60) return `${minutes}m ago`;
    const hours = Math.floor(minutes / 60);
    if (hours < 24) return `${hours}h ago`;
    const days = Math.floor(hours / 24);
    return `${days}d ago`;
  }

  private formatSize(bytes: number): string {
    if (bytes < 1024) return `${bytes} B`;
    if (bytes < 1024 * 1024) return `${(bytes / 1024).toFixed(1)} KB`;
    return `${(bytes / (1024 * 1024)).toFixed(1)} MB`;
  }

  private walkDir(dir: string, callback: (path: string) => void): void {
    if (!fs.existsSync(dir)) return;
    for (const entry of fs.readdirSync(dir, { withFileTypes: true })) {
      const fullPath = path.join(dir, entry.name);
      if (entry.isDirectory()) this.walkDir(fullPath, callback);
      else if (entry.isFile() && entry.name.endsWith('.json')) callback(fullPath);
    }
  }
}
```

### Integration with MarketplaceClient

```typescript
// In MarketplaceClient — wrapper for cached requests
private async cachedRequest<T>(
  cacheKey: string,
  fetcher: () => Promise<T>,
  options?: { ttlMs?: number; forceRefresh?: boolean; forceOffline?: boolean }
): Promise<T> {
  const cache = new CacheManager();

  // Force offline — use cache only
  if (options?.forceOffline) {
    const cached = cache.getStale<T>(cacheKey);
    if (cached) {
      console.log(chalk.dim(`  Using cached data from ${cached.age}`));
      return cached.data;
    }
    throw new Error('No cached data available. Connect to the internet and try again.');
  }

  // Force refresh — skip cache
  if (!options?.forceRefresh) {
    const cached = cache.get<T>(cacheKey);
    if (cached) return cached.data;
  }

  // Fetch from API
  try {
    const data = await fetcher();
    cache.set(cacheKey, data, options?.ttlMs);
    return data;
  } catch (error) {
    // On network error, try stale cache
    if (error instanceof NetworkError) {
      const stale = cache.getStale<T>(cacheKey);
      if (stale) {
        console.log(chalk.yellow(`  Network error. Using cached data from ${stale.age}`));
        return stale.data;
      }
    }
    throw error;
  }
}
```
