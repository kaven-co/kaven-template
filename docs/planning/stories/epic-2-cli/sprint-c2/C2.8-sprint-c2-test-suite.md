## Contexto

Sprint C2 adds five new commands (`init`, `publish`, `upgrade`, `browse`, cache management), enhances `doctor`, implements offline mode, and adds comprehensive documentation. Each feature needs thorough automated testing. This story builds on the test infrastructure established in C1.9 (MSW mock server, JWT helpers, fixture project helpers) and extends it with new test suites for all C2 features.

Testing CLI commands that interact with the filesystem (init), external processes (publish tarball), and interactive prompts (browse) requires specific mocking strategies. This story defines those strategies and ensures all C2 features have reliable automated coverage.

### Prompt Mocking Strategy

Interactive prompts (`@inquirer/prompts`) need mocking for automated tests. Strategy:

```typescript
// tests/helpers/mock-prompts.ts
import * as inquirerPrompts from '@inquirer/prompts';

/**
 * Mock @inquirer/prompts for testing.
 * Provides a queue-based approach: push expected answers, prompts consume them in order.
 */
export class PromptMocker {
  private answers: Array<string | boolean> = [];

  pushAnswers(...answers: Array<string | boolean>): void {
    this.answers.push(...answers);
  }

  install(): void {
    jest.spyOn(inquirerPrompts, 'input').mockImplementation(async () => {
      return this.answers.shift() as string ?? '';
    });
    jest.spyOn(inquirerPrompts, 'select').mockImplementation(async () => {
      return this.answers.shift() as string ?? '';
    });
    jest.spyOn(inquirerPrompts, 'confirm').mockImplementation(async () => {
      return this.answers.shift() as boolean ?? true;
    });
    jest.spyOn(inquirerPrompts, 'password').mockImplementation(async () => {
      return this.answers.shift() as string ?? '';
    });
  }

  restore(): void {
    jest.restoreAllMocks();
  }
}
```

### Init Command Test Example

```typescript
// tests/commands/init/init.test.ts
import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';
import { initCommand } from '../../../src/commands/init/init';
import { PromptMocker } from '../../helpers/mock-prompts';

describe('kaven init', () => {
  let tempDir: string;
  const prompts = new PromptMocker();

  beforeEach(() => {
    tempDir = fs.mkdtempSync(path.join(os.tmpdir(), 'kaven-init-test-'));
    process.chdir(tempDir);
    prompts.install();
  });

  afterEach(() => {
    prompts.restore();
    fs.rmSync(tempDir, { recursive: true, force: true });
  });

  it('rejects invalid project names', async () => {
    const consoleSpy = jest.spyOn(console, 'error').mockImplementation();
    const exitSpy = jest.spyOn(process, 'exit').mockImplementation(() => {
      throw new Error('process.exit');
    });

    await expect(initCommand('My Project', { defaults: true, skipInstall: true, skipGit: true }))
      .rejects.toThrow('process.exit');

    expect(consoleSpy).toHaveBeenCalledWith(
      expect.stringContaining('lowercase letters, numbers, and hyphens')
    );

    consoleSpy.mockRestore();
    exitSpy.mockRestore();
  });

  it('rejects existing directory', async () => {
    fs.mkdirSync(path.join(tempDir, 'existing'));
    const exitSpy = jest.spyOn(process, 'exit').mockImplementation(() => {
      throw new Error('process.exit');
    });

    await expect(initCommand('existing', { defaults: true, skipInstall: true, skipGit: true }))
      .rejects.toThrow('process.exit');

    exitSpy.mockRestore();
  });

  it('replaces placeholders in package.json with --defaults', async () => {
    // Mock git clone to just create a directory with template files
    jest.spyOn(require('child_process'), 'execSync').mockImplementation((cmd: string) => {
      if (cmd.includes('git clone')) {
        const projectDir = path.join(tempDir, 'test-project');
        fs.mkdirSync(projectDir, { recursive: true });
        fs.writeFileSync(
          path.join(projectDir, 'package.json'),
          JSON.stringify({ name: 'kaven-framework', version: '1.0.0' })
        );
        fs.mkdirSync(path.join(projectDir, '.git'));
        fs.writeFileSync(path.join(projectDir, '.env'), 'DATABASE_URL=placeholder\n');
        return Buffer.from('');
      }
      return Buffer.from('');
    });

    await initCommand('test-project', {
      defaults: true,
      skipInstall: true,
      skipGit: true,
    });

    const pkgPath = path.join(tempDir, 'test-project', 'package.json');
    const pkg = JSON.parse(fs.readFileSync(pkgPath, 'utf-8'));
    expect(pkg.name).toBe('test-project');
  });

  it('generates random NEXTAUTH_SECRET', async () => {
    // Similar setup...
    // Verify .env contains non-empty NEXTAUTH_SECRET
    const envContent = fs.readFileSync(
      path.join(tempDir, 'test-project', '.env'), 'utf-8'
    );
    expect(envContent).toMatch(/NEXTAUTH_SECRET="[a-f0-9]{64}"/);
  });
});
```

### CacheManager Test Example

```typescript
// tests/core/CacheManager.test.ts
describe('CacheManager', () => {
  let cacheDir: string;
  let cache: CacheManager;

  beforeEach(() => {
    cacheDir = fs.mkdtempSync(path.join(os.tmpdir(), 'kaven-cache-test-'));
    cache = new CacheManager(cacheDir);
  });

  afterEach(() => fs.rmSync(cacheDir, { recursive: true, force: true }));

  it('stores and retrieves fresh data', () => {
    cache.set('modules:list:page1', { data: [1, 2, 3] });
    const result = cache.get('modules:list:page1');
    expect(result?.data).toEqual({ data: [1, 2, 3] });
  });

  it('returns null for expired entries', () => {
    cache.set('test:expired', { value: 'old' }, 1); // 1ms TTL
    // Wait for expiry
    jest.advanceTimersByTime(100);
    const result = cache.get('test:expired');
    expect(result).toBeNull();
  });

  it('getStale returns expired entries for offline fallback', () => {
    cache.set('test:stale', { value: 'stale' }, 1);
    jest.advanceTimersByTime(100);
    const result = cache.getStale('test:stale');
    expect(result?.data).toEqual({ value: 'stale' });
  });

  it('evicts LRU entries when over size limit', () => {
    // Create entries totaling > 50MB
    const bigData = 'x'.repeat(10 * 1024 * 1024); // 10MB per entry
    for (let i = 0; i < 6; i++) {
      cache.set(`test:big:${i}`, bigData);
      // Simulate different access times
    }

    // After eviction, total should be <=40MB (80% of 50MB)
    const status = cache.getStatus();
    expect(parseInt(status.size)).toBeLessThanOrEqual(50);
  });

  it('clear removes all cached data', () => {
    cache.set('test:a', 'data1');
    cache.set('test:b', 'data2');
    cache.clear();
    expect(cache.get('test:a')).toBeNull();
    expect(cache.get('test:b')).toBeNull();
  });
});
```

```typescript
describe('kaven doctor', () => {
  it('returns exit code 0 when all checks pass', async () => {
    const project = createFixtureProject();
```
