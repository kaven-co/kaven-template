# C2.1 — `kaven init` Command

> **Epic:** 2 (CLI) | **Sprint:** C2 | **Agent:** @dev
> **Priority:** P0
> **Depends on:** None (standalone new command)

---

## Contexto

The `kaven init` command is the primary entry point for new Kaven users. It bootstraps a new Kaven project by cloning the kaven-framework repository, running interactive setup prompts, replacing placeholder values in configuration files, and executing initial setup commands (dependency install, Prisma generation, database migration).

A key design decision: demos stay intact. The cloned project includes working example modules (projects, tasks) so the developer can see a real multi-tenant SaaS application running before they customize it. The README explains how to remove demo modules when ready.

## User Story

**As a** developer starting a new SaaS project,
**I want to** run a single command to bootstrap a complete Kaven project,
**So that** I can have a working multi-tenant application running in minutes instead of days.

## Acceptance Criteria

- [ ] `kaven init <project-name>` creates a new directory with the given name
- [ ] If directory already exists: show error `"Directory '<name>' already exists. Choose a different name."`
- [ ] Clones `kaven-framework` repository (from GitHub, via `git clone --depth 1`)
- [ ] Clone URL configurable via `KAVEN_TEMPLATE_URL` env var (default: `https://github.com/kaven-co/kaven-framework.git`)
- [ ] Removes `.git` directory after clone (fresh start)
- [ ] Interactive prompts collect:
  - Project name (pre-filled from CLI argument)
  - Database URL (default: `postgresql://localhost:5432/<project-name>`)
  - Admin email (required, validated as email format)
  - Default locale (choices: `en-US`, `pt-BR`, `es-ES`, default: `en-US`)
  - Default currency (choices: `USD`, `BRL`, `EUR`, default: `USD`)
- [ ] `--defaults` flag skips prompts and uses all defaults
- [ ] Placeholder replacement in files:
  - `package.json`: `name` field → project name
  - `.env`: `DATABASE_URL` → user's database URL
  - `.env`: `ADMIN_EMAIL` → user's admin email
  - `.env`: `DEFAULT_LOCALE` → chosen locale
  - `.env`: `DEFAULT_CURRENCY` → chosen currency
  - `prisma/schema.prisma`: database URL reference
- [ ] Demo modules stay intact (projects, tasks modules as working examples)
- [ ] Runs setup commands sequentially with progress:
  1. `pnpm install` (with spinner)
  2. `npx prisma generate` (with spinner)
  3. `npx prisma migrate dev --name init` (with spinner)
- [ ] If any setup command fails: show error but don't delete project (user can fix manually)
- [ ] Initializes new git repository: `git init && git add . && git commit -m "Initial commit from Kaven"`
- [ ] Success message with next steps:
  ```
  Project 'my-saas' created successfully!

  Next steps:
    cd my-saas
    pnpm dev              # Start development server
    open http://localhost:3000   # Admin panel
    open http://localhost:3001   # Tenant app

  To remove demo modules:
    kaven module remove projects
    kaven module remove tasks
  ```
- [ ] `--skip-install` flag skips pnpm install and Prisma commands
- [ ] `--skip-git` flag skips git init

## Technical Notes

### Command Registration

```typescript
// src/commands/init/index.ts
import { Command } from 'commander';

export function registerInitCommand(program: Command) {
  program
    .command('init <project-name>')
    .description('Create a new Kaven project')
    .option('--defaults', 'Use default values for all prompts')
    .option('--skip-install', 'Skip dependency installation and Prisma setup')
    .option('--skip-git', 'Skip git initialization')
    .option('--template-url <url>', 'Custom template repository URL')
    .action(async (projectName, opts) => {
      await initCommand(projectName, opts);
    });
}
```

### Init Implementation

```typescript
// src/commands/init/init.ts
import { execSync, spawn } from 'child_process';
import * as fs from 'fs';
import * as path from 'path';
import chalk from 'chalk';
import ora from 'ora';
import { input, select } from '@inquirer/prompts';

interface InitOptions {
  defaults?: boolean;
  skipInstall?: boolean;
  skipGit?: boolean;
  templateUrl?: string;
}

interface ProjectConfig {
  name: string;
  databaseUrl: string;
  adminEmail: string;
  locale: string;
  currency: string;
}

export async function initCommand(projectName: string, options: InitOptions) {
  const projectDir = path.resolve(process.cwd(), projectName);

  // Validate project name
  if (!/^[a-z0-9-]+$/.test(projectName)) {
    console.error(chalk.red('Project name must contain only lowercase letters, numbers, and hyphens.'));
    process.exit(1);
  }

  // Check if directory exists
  if (fs.existsSync(projectDir)) {
    console.error(chalk.red(`Directory '${projectName}' already exists. Choose a different name.`));
    process.exit(1);
  }

  console.log(chalk.bold(`\n  Creating Kaven project: ${projectName}\n`));

  // Gather config
  const config = options.defaults
    ? getDefaults(projectName)
    : await promptConfig(projectName);

  // Step 1: Clone template
  const spinner = ora('Cloning Kaven framework...').start();
  const templateUrl = options.templateUrl
    ?? process.env.KAVEN_TEMPLATE_URL
    ?? 'https://github.com/kaven-co/kaven-framework.git';

  try {
    execSync(`git clone --depth 1 "${templateUrl}" "${projectDir}"`, {
      stdio: 'pipe',
    });
    // Remove .git (fresh start)
    fs.rmSync(path.join(projectDir, '.git'), { recursive: true, force: true });
    spinner.succeed('Template cloned');
  } catch (error) {
    spinner.fail('Failed to clone template');
    console.error(chalk.red(`\n  Could not clone from ${templateUrl}`));
    console.error(chalk.dim('  Check your internet connection and try again.'));
    process.exit(1);
  }

  // Step 2: Replace placeholders
  spinner.start('Configuring project...');
  replacePlaceholders(projectDir, config);
  spinner.succeed('Project configured');

  // Step 3: Install dependencies
  if (!options.skipInstall) {
    const setupSteps = [
      { label: 'Installing dependencies...', cmd: 'pnpm install' },
      { label: 'Generating Prisma client...', cmd: 'npx prisma generate' },
      { label: 'Running database migration...', cmd: 'npx prisma migrate dev --name init' },
    ];

    for (const step of setupSteps) {
      spinner.start(step.label);
      try {
        execSync(step.cmd, { cwd: projectDir, stdio: 'pipe' });
        spinner.succeed(step.label.replace('...', ''));
      } catch (error) {
        spinner.warn(`${step.label.replace('...', '')} failed — run manually later`);
      }
    }
  }

  // Step 4: Git init
  if (!options.skipGit) {
    spinner.start('Initializing git repository...');
    try {
      execSync('git init', { cwd: projectDir, stdio: 'pipe' });
      execSync('git add .', { cwd: projectDir, stdio: 'pipe' });
      execSync('git commit -m "Initial commit from Kaven"', { cwd: projectDir, stdio: 'pipe' });
      spinner.succeed('Git repository initialized');
    } catch {
      spinner.warn('Git initialization failed — run manually');
    }
  }

  // Success message
  console.log(chalk.green.bold(`\n  ✓ Project '${projectName}' created successfully!\n`));
  console.log(chalk.bold('  Next steps:'));
  console.log(chalk.dim(`    cd ${projectName}`));
  console.log(chalk.dim('    pnpm dev              # Start development server'));
  console.log(chalk.dim('    open http://localhost:3000   # Admin panel'));
  console.log(chalk.dim('    open http://localhost:3001   # Tenant app'));
  console.log();
  console.log(chalk.bold('  To remove demo modules:'));
  console.log(chalk.dim('    kaven module remove projects'));
  console.log(chalk.dim('    kaven module remove tasks'));
  console.log();
}

async function promptConfig(projectName: string): Promise<ProjectConfig> {
  const name = await input({
    message: 'Project name:',
    default: projectName,
    validate: (v) => /^[a-z0-9-]+$/.test(v) || 'Lowercase letters, numbers, and hyphens only',
  });

  const databaseUrl = await input({
    message: 'Database URL:',
    default: `postgresql://localhost:5432/${name}`,
  });

  const adminEmail = await input({
    message: 'Admin email:',
    validate: (v) => /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(v) || 'Enter a valid email',
  });

  const locale = await select({
    message: 'Default locale:',
    choices: [
      { value: 'en-US', name: 'English (US)' },
      { value: 'pt-BR', name: 'Portugues (Brasil)' },
      { value: 'es-ES', name: 'Espanol (Espana)' },
    ],
    default: 'en-US',
  });

  const currency = await select({
    message: 'Default currency:',
    choices: [
      { value: 'USD', name: 'USD - US Dollar' },
      { value: 'BRL', name: 'BRL - Real Brasileiro' },
      { value: 'EUR', name: 'EUR - Euro' },
    ],
    default: 'USD',
  });

  return { name, databaseUrl, adminEmail, locale, currency };
}

function getDefaults(projectName: string): ProjectConfig {
  return {
    name: projectName,
    databaseUrl: `postgresql://localhost:5432/${projectName}`,
    adminEmail: 'admin@localhost',
    locale: 'en-US',
    currency: 'USD',
  };
}

function replacePlaceholders(projectDir: string, config: ProjectConfig): void {
  // package.json — update name
  const pkgPath = path.join(projectDir, 'package.json');
  if (fs.existsSync(pkgPath)) {
    const pkg = JSON.parse(fs.readFileSync(pkgPath, 'utf-8'));
    pkg.name = config.name;
    fs.writeFileSync(pkgPath, JSON.stringify(pkg, null, 2) + '\n');
  }

  // .env — create or update
  const envPath = path.join(projectDir, '.env');
  const envContent = [
    `# Generated by kaven init`,
    `DATABASE_URL="${config.databaseUrl}"`,
    `ADMIN_EMAIL="${config.adminEmail}"`,
    `DEFAULT_LOCALE="${config.locale}"`,
    `DEFAULT_CURRENCY="${config.currency}"`,
    `NEXTAUTH_SECRET="${generateRandomSecret()}"`,
    `NEXTAUTH_URL="http://localhost:3000"`,
    '',
  ].join('\n');

  // Merge with existing .env template if present
  if (fs.existsSync(envPath)) {
    const existing = fs.readFileSync(envPath, 'utf-8');
    const merged = mergeEnv(existing, envContent);
    fs.writeFileSync(envPath, merged);
  } else {
    fs.writeFileSync(envPath, envContent);
  }
}

function generateRandomSecret(): string {
  const { randomBytes } = require('crypto');
  return randomBytes(32).toString('hex');
}

function mergeEnv(existing: string, updates: string): string {
  const existingVars = new Map<string, string>();
  for (const line of existing.split('\n')) {
    const match = line.match(/^([A-Z_]+)=(.*)$/);
    if (match) existingVars.set(match[1], match[2]);
  }

  for (const line of updates.split('\n')) {
    const match = line.match(/^([A-Z_]+)=(.*)$/);
    if (match) existingVars.set(match[1], match[2]);
  }

  return Array.from(existingVars.entries())
    .map(([k, v]) => `${k}=${v}`)
    .join('\n') + '\n';
}
```
