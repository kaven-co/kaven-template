# M2.3 — Artifact Storage (S3)

> **Epic:** 1 (Marketplace) | **Sprint:** M2 | **Agent:** @devops
> **Priority:** P0
> **Depends on:** (none)

---

## Contexto

Module artifacts (tar.gz bundles) need a reliable, scalable storage backend. AWS S3 is the industry standard for object storage and provides the durability, availability, and presigned URL capabilities we need.

Publishers upload artifacts via presigned PUT URLs (avoiding our server as a bottleneck), and consumers download via the download token system (M2.2) which proxies from S3. Artifact integrity is guaranteed by Ed25519 signatures (generated at publish time, verified at install time) and SHA-256 checksums.

Local development uses MinIO, an S3-compatible object store, to avoid AWS costs and enable offline development.

## User Story

**Como** publisher de módulos, **Quero** fazer upload de artefatos de forma segura para o marketplace, **Para que** consumidores possam baixar módulos verificados e íntegros.

## Acceptance Criteria

- [ ] StorageService.upload() uploads a file buffer to S3 at the correct key path
- [ ] S3 bucket key structure follows: `{moduleSlug}/{version}/module.tar.gz`
- [ ] StorageService.getPresignedUploadUrl() generates a PUT presigned URL (1 hour TTL)
- [ ] StorageService.getPresignedDownloadUrl() generates a GET presigned URL (15 min TTL)
- [ ] StorageService.download() returns a readable stream from S3
- [ ] StorageService.delete() removes an artifact from S3
- [ ] Ed25519 signature is generated at upload time and stored alongside the artifact
- [ ] Signature file stored at: `{moduleSlug}/{version}/module.tar.gz.sig`
- [ ] StorageService.verifySignature() validates artifact against stored signature
- [ ] SHA-256 checksum is computed and stored in release metadata
- [ ] Checksum verification happens before completing download to consumer
- [ ] All S3 operations use the configured bucket from `S3_BUCKET_NAME` env var
- [ ] MinIO docker-compose service is configured for local development
- [ ] S3 client uses `S3_ENDPOINT` env var (MinIO URL locally, omitted for real AWS)
- [ ] Upload fails gracefully with meaningful error if S3 is unreachable
- [ ] File size limit enforced: 50MB max per artifact

## Technical Notes

### StorageService Implementation

```typescript
// src/services/storage.service.ts

import {
  S3Client,
  PutObjectCommand,
  GetObjectCommand,
  DeleteObjectCommand,
} from '@aws-sdk/client-s3';
import { getSignedUrl } from '@aws-sdk/s3-request-presigner';
import crypto from 'node:crypto';
import { Readable } from 'node:stream';

const MAX_ARTIFACT_SIZE = 50 * 1024 * 1024; // 50MB

export class StorageService {
  private s3: S3Client;
  private bucket: string;
  private signingKeyPair: { publicKey: Buffer; privateKey: Buffer };

  constructor(config: StorageConfig) {
    this.s3 = new S3Client({
      region: config.region || 'us-east-1',
      ...(config.endpoint && {
        endpoint: config.endpoint,
        forcePathStyle: true, // Required for MinIO
      }),
      credentials: {
        accessKeyId: config.accessKeyId,
        secretAccessKey: config.secretAccessKey,
      },
    });
    this.bucket = config.bucket;
    this.signingKeyPair = config.signingKeyPair;
  }

  private artifactKey(moduleSlug: string, version: string): string {
    return `${moduleSlug}/${version}/module.tar.gz`;
  }

  private signatureKey(moduleSlug: string, version: string): string {
    return `${moduleSlug}/${version}/module.tar.gz.sig`;
  }

  async upload(
    moduleSlug: string,
    version: string,
    buffer: Buffer
  ): Promise<UploadResult> {
    if (buffer.length > MAX_ARTIFACT_SIZE) {
      throw new PayloadTooLargeError(
        `Artifact size ${buffer.length} exceeds limit of ${MAX_ARTIFACT_SIZE}`
      );
    }

    const key = this.artifactKey(moduleSlug, version);

    // Compute SHA-256 checksum
    const checksum = crypto.createHash('sha256').update(buffer).digest('hex');

    // Generate Ed25519 signature
    const signature = crypto.sign(null, buffer, {
      key: this.signingKeyPair.privateKey,
      format: 'pem',
      type: 'pkcs8',
    });

    // Upload artifact
    await this.s3.send(
      new PutObjectCommand({
        Bucket: this.bucket,
        Key: key,
        Body: buffer,
        ContentType: 'application/gzip',
        Metadata: {
          'x-kaven-checksum': checksum,
          'x-kaven-module': moduleSlug,
          'x-kaven-version': version,
        },
      })
    );

    // Upload signature
    await this.s3.send(
      new PutObjectCommand({
        Bucket: this.bucket,
        Key: this.signatureKey(moduleSlug, version),
        Body: signature,
        ContentType: 'application/octet-stream',
      })
    );

    return { key, checksum, signatureKey: this.signatureKey(moduleSlug, version) };
  }

  async download(key: string): Promise<Readable> {
    const response = await this.s3.send(
      new GetObjectCommand({
        Bucket: this.bucket,
        Key: key,
      })
    );

    if (!response.Body) {
      throw new NotFoundError(`Artifact not found: ${key}`);
    }

    return response.Body as Readable;
  }

  async getPresignedUploadUrl(
    moduleSlug: string,
    version: string
  ): Promise<{ url: string; expiresAt: Date }> {
    const command = new PutObjectCommand({
      Bucket: this.bucket,
      Key: this.artifactKey(moduleSlug, version),
      ContentType: 'application/gzip',
    });

    const url = await getSignedUrl(this.s3, command, { expiresIn: 3600 }); // 1 hour
    const expiresAt = new Date(Date.now() + 3600 * 1000);

    return { url, expiresAt };
  }

  async getPresignedDownloadUrl(
    moduleSlug: string,
    version: string
  ): Promise<{ url: string; expiresAt: Date }> {
    const command = new GetObjectCommand({
      Bucket: this.bucket,
      Key: this.artifactKey(moduleSlug, version),
    });

    const url = await getSignedUrl(this.s3, command, { expiresIn: 900 }); // 15 min
    const expiresAt = new Date(Date.now() + 900 * 1000);

    return { url, expiresAt };
  }

  async verifySignature(
    moduleSlug: string,
    version: string,
    artifactBuffer: Buffer
  ): Promise<boolean> {
    const sigResponse = await this.s3.send(
      new GetObjectCommand({
        Bucket: this.bucket,
        Key: this.signatureKey(moduleSlug, version),
      })
    );

    if (!sigResponse.Body) return false;

    const chunks: Buffer[] = [];
    for await (const chunk of sigResponse.Body as Readable) {
      chunks.push(Buffer.from(chunk));
    }
    const signature = Buffer.concat(chunks);

    return crypto.verify(null, artifactBuffer, {
      key: this.signingKeyPair.publicKey,
      format: 'pem',
      type: 'spki',
    }, signature);
  }

  async delete(moduleSlug: string, version: string): Promise<void> {
    await Promise.all([
      this.s3.send(
        new DeleteObjectCommand({
          Bucket: this.bucket,
          Key: this.artifactKey(moduleSlug, version),
        })
      ),
      this.s3.send(
        new DeleteObjectCommand({
          Bucket: this.bucket,
          Key: this.signatureKey(moduleSlug, version),
        })
      ),
    ]);
  }
}
```

### Docker Compose Addition (MinIO)

```yaml
# docker-compose.yml addition

services:
  minio:
    image: minio/minio:latest
    container_name: kaven-minio
    ports:
      - '9000:9000'    # S3 API
      - '9001:9001'    # MinIO Console
    environment:
      MINIO_ROOT_USER: kaven-dev
      MINIO_ROOT_PASSWORD: kaven-dev-secret
    volumes:
      - minio-data:/data
    command: server /data --console-address ":9001"
    healthcheck:
      test: ['CMD', 'mc', 'ready', 'local']
      interval: 10s
      timeout: 5s
      retries: 5

  minio-init:
    image: minio/mc:latest
    container_name: kaven-minio-init
    depends_on:
      minio:
        condition: service_healthy
    entrypoint: >
      /bin/sh -c "
        mc alias set kaven http://minio:9000 kaven-dev kaven-dev-secret &&
        mc mb --ignore-existing kaven/kaven-marketplace-artifacts &&
        echo 'Bucket created successfully'
      "

volumes:
  minio-data:
```
