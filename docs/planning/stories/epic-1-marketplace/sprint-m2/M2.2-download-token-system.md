# M2.2 — Download Token System

> **Epic:** 1 (Marketplace) | **Sprint:** M2 | **Agent:** @dev
> **Priority:** P0
> **Depends on:** M2.1

---

## Contexto

Direct download URLs are a security liability — they can be shared, bookmarked, and reused without authorization. The download token system creates an ephemeral, single-use abstraction between license validation and artifact delivery.

When a user requests a module download, the system validates their license (M2.1), generates a short-lived token (15 minutes), and returns a download URL containing that token. The artifact endpoint (GET /artifacts/:token) streams the file from S3 and immediately marks the token as consumed. This prevents unauthorized redistribution while keeping the download UX frictionless.

The token also ties downloads to specific releases, enabling accurate download counting per module and per version.

## User Story

**Como** developer autenticado com licença válida, **Quero** receber um link temporário para baixar módulos, **Para que** apenas downloads autorizados sejam possíveis e eu não possa compartilhar links permanentes.

---

## Acceptance Criteria

- [ ] `POST /download-tokens` requires authentication (JWT)
- [ ] `POST /download-tokens` accepts `{ moduleSlug: string, version?: string }` (version defaults to latest)
- [ ] Token generation validates: user has active license, license tier satisfies module requirement
- [ ] Generated token is a cryptographically random 64-char hex string (32 bytes)
- [ ] Token TTL is 15 minutes from creation
- [ ] Response includes `{ token: string, downloadUrl: string, expiresAt: Date }`
- [ ] `GET /artifacts/:token` streams the artifact file from S3
- [ ] `GET /artifacts/:token` sets correct headers: `Content-Type: application/gzip`, `Content-Disposition: attachment`
- [ ] Token is single-use: `usedAt` is set after successful download
- [ ] Attempting to reuse a consumed token returns 410 Gone
- [ ] Attempting to use an expired token returns 410 Gone
- [ ] Successful download increments `Module.installCount` atomically
- [ ] Successful download creates a `DownloadLog` entry with userId, moduleId, releaseId, timestamp
- [ ] Token generation for a module with `requiredTier: null` (free) still requires authentication but no license check
- [ ] Invalid moduleSlug returns 404
- [ ] Invalid version returns 404

### New Prisma Model

```prisma
model DownloadToken {
  id         String    @id @default(cuid())
  token      String    @unique /// @indexed — lookup on every download
  userId     String
  moduleId   String
  releaseId  String
  expiresAt  DateTime
  usedAt     DateTime?
  createdAt  DateTime  @default(now())

  user       User      @relation(fields: [userId], references: [id])
  module     Module    @relation(fields: [moduleId], references: [id])
  release    Release   @relation(fields: [releaseId], references: [id])

  @@index([token])
  @@index([userId, createdAt])
  @@index([expiresAt]) /// For cleanup cron
}

model DownloadLog {
  id         String   @id @default(cuid())
  userId     String
  moduleId   String
  releaseId  String
  version    String
  ipAddress  String?
  userAgent  String?

  user       User     @relation(fields: [userId], references: [id])
  module     Module   @relation(fields: [moduleId], references: [id])
  release    Release  @relation(fields: [releaseId], references: [id])

  @@index([moduleId, createdAt])
  @@index([userId, createdAt])
```

### DownloadTokenService

```typescript
// src/services/download-token.service.ts

import crypto from 'node:crypto';

const TOKEN_TTL_MS = 15 * 60 * 1000; // 15 minutes

export class DownloadTokenService {
  constructor(
    private prisma: PrismaClient,
    private licenseService: LicenseService,
    private storageService: StorageService
  ) {}

  async createToken(userId: string, moduleSlug: string, version?: string): Promise<TokenResponse> {
    // 1. Find module
    const module = await this.prisma.module.findUnique({
      where: { slug: moduleSlug },
      include: { releases: { orderBy: { createdAt: 'desc' }, take: 1 } },
    });
    if (!module) throw new NotFoundError('Module not found');

    // 2. Resolve release version
    const release = version
      ? await this.prisma.release.findFirst({
          where: { moduleId: module.id, version },
        })
      : module.releases[0];
    if (!release) throw new NotFoundError('Release not found');

    // 3. Validate license (skip for free modules)
    if (module.requiredTier !== null) {
      const licenses = await this.licenseService.getByUser(userId, { status: 'ACTIVE' });
      const validLicense = licenses.find(
        (l) => tierSatisfiesRequirement(l.type, module.requiredTier)
      );
      if (!validLicense) {
        throw new ForbiddenError('No valid license for this module tier');
      }
    }

    // 4. Generate token
    const token = crypto.randomBytes(32).toString('hex');
    const expiresAt = new Date(Date.now() + TOKEN_TTL_MS);

    const downloadToken = await this.prisma.downloadToken.create({
      data: {
        token,
        userId,
        moduleId: module.id,
        releaseId: release.id,
        expiresAt,
      },
    });

    return {
      token: downloadToken.token,
      downloadUrl: `/artifacts/${downloadToken.token}`,
      expiresAt: downloadToken.expiresAt,
    };
  }

  async consumeToken(token: string): Promise<ArtifactStream> {
    const downloadToken = await this.prisma.downloadToken.findUnique({
      where: { token },
      include: { module: true, release: true },
    });

    if (!downloadToken) throw new NotFoundError('Token not found');
    if (downloadToken.usedAt) throw new GoneError('Token already consumed');
    if (downloadToken.expiresAt < new Date()) throw new GoneError('Token expired');

    // Mark as used and increment install count atomically
    const [_, updatedModule] = await this.prisma.$transaction([
      this.prisma.downloadToken.update({
        where: { id: downloadToken.id },
        data: { usedAt: new Date() },
      }),
      this.prisma.module.update({
        where: { id: downloadToken.moduleId },
        data: { installCount: { increment: 1 } },
      }),
    ]);

    // Create download log (fire and forget)
    this.prisma.downloadLog.create({
      data: {
        userId: downloadToken.userId,
        moduleId: downloadToken.moduleId,
        releaseId: downloadToken.releaseId,
        version: downloadToken.release.version,
      },
    }).catch((err) => console.error('Failed to create download log:', err));

    // Get artifact stream from S3
    const stream = await this.storageService.download(
      `${downloadToken.module.slug}/${downloadToken.release.version}/module.tar.gz`
    );

    return {
      stream,
      filename: `${downloadToken.module.slug}-${downloadToken.release.version}.tar.gz`,
      contentType: 'application/gzip',
    };
  }
}
```

### Route Definitions

```typescript
// src/routes/download-tokens.route.ts

const downloadTokensRoute: FastifyPluginAsync = async (fastify) => {
  // POST /download-tokens — generate token
  fastify.post('/', {
    preHandler: [fastify.authenticate],
    schema: {
      body: {
        type: 'object',
        required: ['moduleSlug'],
        properties: {
          moduleSlug: { type: 'string', minLength: 1 },
          version: { type: 'string', pattern: '^\\d+\\.\\d+\\.\\d+' },
        },
      },
    },
    handler: async (request, reply) => {
      const { moduleSlug, version } = request.body as any;
      const result = await downloadTokenService.createToken(
        request.user.id, moduleSlug, version
      );
      return reply.code(201).send(result);
    },
  });

  // GET /artifacts/:token — download artifact
  fastify.get('/artifacts/:token', {
    schema: {
      params: {
        type: 'object',
        required: ['token'],
        properties: {
          token: { type: 'string', minLength: 64, maxLength: 64 },
        },
      },
    },
    handler: async (request, reply) => {
      const { token } = request.params as any;
      const artifact = await downloadTokenService.consumeToken(token);

      reply
        .header('Content-Type', artifact.contentType)
        .header('Content-Disposition', `attachment; filename="${artifact.filename}"`)
        .send(artifact.stream);
    },
  });
};
```

### Expired Token Cleanup (Cron)

```typescript
// src/jobs/cleanup-expired-tokens.ts
// Run every hour via node-cron or Fastify plugin

async function cleanupExpiredTokens(prisma: PrismaClient): Promise<number> {
  const { count } = await prisma.downloadToken.deleteMany({
    where: {
      OR: [
        { expiresAt: { lt: new Date() } },
        { usedAt: { not: null }, createdAt: { lt: new Date(Date.now() - 24 * 60 * 60 * 1000) } },
      ],
    },
  });
  return count;
}
```
