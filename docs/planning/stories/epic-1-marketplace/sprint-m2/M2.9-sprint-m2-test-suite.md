# M2.9 — Sprint M2 Test Suite

> **Epic:** 1 (Marketplace) | **Sprint:** M2 | **Agent:** @qa
> **Priority:** P0
> **Depends on:** M2.1, M2.2, M2.3, M2.4, M2.5, M2.6, M2.7, M2.8

---

## Contexto

Sprint M2 introduces critical payment and licensing infrastructure. A single bug in license validation could lock paying users out of modules. A flaw in webhook handling could create duplicate licenses or miss payments entirely. A cache invalidation failure could serve stale license data after revocation.

This story establishes comprehensive test coverage for every M2 component. Tests use Vitest (consistent with the rest of kaven-framework), with mocked external services (Paddle API, AWS S3, Redis) to enable fast, deterministic testing without real infrastructure.

Target: 80%+ code coverage on all new M2 code, with 50+ individual test cases covering happy paths, error paths, edge cases, and integration scenarios.

## User Story

**Como** QA engineer, **Quero** uma suite de testes abrangente para todo o Sprint M2, **Para que** regressoes em licensing, downloads e pagamentos sejam detectadas imediatamente.

## Acceptance Criteria

- [ ] License CRUD tests: create, getByUser, getByKey, validate (active), validate (expired), validate (revoked), validate (wrong tier), revoke, revoke already-revoked (409)
- [ ] License tier hierarchy tests: PRO accesses STARTER module, STARTER cannot access PRO module, null requiredTier accessible by all
- [ ] Download token tests: create token (happy path), create with invalid license, create for free module (no license needed), consume token, consume expired token (410), consume already-used token (410), token format validation
- [ ] S3 storage tests: upload artifact, download artifact, generate presigned URLs, verify Ed25519 signature, reject tampered artifact, delete artifact, handle S3 error gracefully
- [ ] Paddle webhook tests: signature verification (valid), signature verification (invalid → 403), subscription_created → license created, transaction_completed → license activated, subscription_updated → tier changed, subscription_cancelled → license expired, transaction_refunded → license revoked, duplicate event (idempotency), unknown event type (accepted, logged)
- [ ] Checkout tests: create session (happy path), create session with existing higher tier (409), verify transaction status, portal URL generation
- [ ] License key tests: generate valid key, validate correct key, reject wrong prefix, reject invalid tier code, reject bad characters, reject bad checksum, parse extracts tier correctly, generateUnique retries on collision
- [ ] Rate limit tests: auth endpoint hits limit at 10 requests, device poll returns slow_down, module listing limit, download limit, webhook has no limit, 429 includes Retry-After header
- [ ] Redis cache tests: cache hit returns cached data, cache miss queries DB, invalidate removes entry, invalidatePattern removes matching entries, Redis down bypasses cache gracefully, TTL expiry works
- [ ] Integration tests: full flow (checkout → webhook → license created → download token → artifact download)
- [ ] All tests pass in CI (no flaky tests)
- [ ] Coverage report generated: 80%+ on M2 code

import { LicenseKeyService } from '../../../src/services/license-key.service';

    it('should generate key with correct format', () => {
      const key = service.generate('PRO');
      expect(key).toMatch(/^KAVEN-PRO-[ABCDEFGHJKMNPQRSTUVWXYZ23456789]{8}-[ABCDEFGHJKMNPQRSTUVWXYZ23456789]{2}$/);
    });

    it('should generate key with correct tier code for each tier', () => {
      expect(service.generate('STARTER')).toContain('-STR-');
      expect(service.generate('COMPLETE')).toContain('-CMP-');
      expect(service.generate('PRO')).toContain('-PRO-');
      expect(service.generate('ENTERPRISE')).toContain('-ENT-');
    });

    it('should generate unique keys (statistical test)', () => {
      const keys = new Set(Array.from({ length: 100 }, () => service.generate('PRO')));
      expect(keys.size).toBe(100);
    });

    it('should not include confusing characters', () => {
      const keys = Array.from({ length: 50 }, () => service.generate('PRO'));
      for (const key of keys) {
        const randomPart = key.split('-')[2];
        expect(randomPart).not.toMatch(/[0OIL1]/);
      }
    });
  });

  describe('validate', () => {
    it('should validate a correctly generated key', () => {
      const key = service.generate('PRO');
      expect(service.validate(key)).toEqual({ valid: true });
    });

    it('should reject key with wrong prefix', () => {
      const result = service.validate('NOTKA-PRO-ABCDEFGH-XX');
      expect(result.valid).toBe(false);
      expect(result.reason).toBe('INVALID_FORMAT');
    });

    it('should reject key with invalid tier code', () => {
      const result = service.validate('KAVEN-XXX-ABCDEFGH-XX');
      expect(result.valid).toBe(false);
      expect(result.reason).toBe('INVALID_FORMAT');
    });

    it('should reject key with confusing characters', () => {
      const result = service.validate('KAVEN-PRO-A0O1ILBC-XX');
      expect(result.valid).toBe(false);
    });

    it('should reject key with bad checksum', () => {
      const key = service.generate('PRO');
      const tampered = key.slice(0, -2) + 'AA'; // Replace checksum
      const result = service.validate(tampered);
      expect(result.valid).toBe(false);
      expect(result.reason).toBe('INVALID_CHECKSUM');
    });

    it('should reject lowercase keys', () => {
      const result = service.validate('kaven-pro-abcdefgh-xx');
      expect(result.valid).toBe(false);
      expect(result.reason).toBe('INVALID_FORMAT');
    });
  });

  describe('parse', () => {
    it('should extract tier from key', () => {
      const key = service.generate('COMPLETE');
      const parsed = service.parse(key);
      expect(parsed.tier).toBe('COMPLETE');
      expect(parsed.tierCode).toBe('CMP');
    });

    it('should throw on invalid key', () => {
      expect(() => service.parse('INVALID-KEY')).toThrow();
    });
  });
});

// tests/unit/utils/paddle-signature.test.ts

import { describe, it, expect } from 'vitest';
import crypto from 'node:crypto';
import { verifyPaddleWebhookSignature } from '../../../src/utils/paddle-signature';

describe('verifyPaddleWebhookSignature', () => {
  const secret = 'test-webhook-secret';

  function createValidSignature(body: string, ts: string): string {
    const signedPayload = `${ts}:${body}`;
    const hash = crypto.createHmac('sha256', secret).update(signedPayload).digest('hex');
    return `ts=${ts};h1=${hash}`;
  }

  it('should verify valid signature', () => {
    const body = '{"event_type":"test"}';
    const ts = '1707000000';
    const signature = createValidSignature(body, ts);

    expect(verifyPaddleWebhookSignature(body, signature, secret)).toBe(true);
  });

  it('should reject invalid signature', () => {
    const body = '{"event_type":"test"}';
    const signature = 'ts=1707000000;h1=invalidsignaturehex';

    expect(verifyPaddleWebhookSignature(body, signature, secret)).toBe(false);
  });

  it('should reject tampered body', () => {
    const body = '{"event_type":"test"}';
    const ts = '1707000000';
    const signature = createValidSignature(body, ts);

    expect(verifyPaddleWebhookSignature('{"event_type":"tampered"}', signature, secret)).toBe(false);
  });

  it('should reject missing signature parts', () => {
    expect(verifyPaddleWebhookSignature('body', 'invalid', secret)).toBe(false);
    expect(verifyPaddleWebhookSignature('body', 'ts=123', secret)).toBe(false);
    expect(verifyPaddleWebhookSignature('body', 'h1=abc', secret)).toBe(false);
  });
});

// tests/integration/rate-limit.test.ts

import { describe, it, expect, beforeAll, afterAll } from 'vitest';
import { buildTestApp } from '../helpers/test-app';

describe('Rate Limiting', () => {
  let app: FastifyInstance;

  beforeAll(async () => {
    app = await buildTestApp();
  });

  afterAll(async () => {
    await app.close();
  });

  it('should return 429 after exceeding auth rate limit', async () => {
    const requests = Array.from({ length: 12 }, () =>
      app.inject({ method: 'POST', url: '/auth/login', payload: { email: 'test@test.com', password: 'wrong' } })
    );

    const responses = await Promise.all(requests);
    const tooManyRequests = responses.filter((r) => r.statusCode === 429);
    expect(tooManyRequests.length).toBeGreaterThanOrEqual(1);
  });

  it('should include Retry-After header on 429', async () => {
    // Exhaust limit first
    for (let i = 0; i < 11; i++) {
      await app.inject({ method: 'POST', url: '/auth/login', payload: {} });
    }

    const response = await app.inject({ method: 'POST', url: '/auth/login', payload: {} });
    if (response.statusCode === 429) {
      expect(response.headers['retry-after']).toBeDefined();
      const body = JSON.parse(response.body);
      expect(body.error).toBe('TOO_MANY_REQUESTS');
      expect(body.retryAfter).toBeGreaterThan(0);
    }
  });

  it('should not rate limit webhook endpoints', async () => {
    const requests = Array.from({ length: 200 }, () =>
      app.inject({ method: 'POST', url: '/webhooks/paddle', payload: {}, headers: { 'paddle-signature': 'invalid' } })
    );

    const responses = await Promise.all(requests);
    const rateLimited = responses.filter((r) => r.statusCode === 429);
    expect(rateLimited.length).toBe(0);
  });
});
