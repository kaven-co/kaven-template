# M2.8 — Redis Caching Layer

> **Epic:** 1 (Marketplace) | **Sprint:** M2 | **Agent:** @dev
> **Priority:** P1
> **Depends on:** M2.1

---

## Contexto

Database queries are the primary bottleneck of any API. The marketplace module listing endpoint, license validation, and Paddle product mapping are all read-heavy operations that benefit massively from caching. A single module listing page might be requested hundreds of times per minute with identical results.

Redis provides sub-millisecond reads, making it ideal for caching these hot paths. The caching layer is designed as a service that wraps all cache operations, with consistent key naming, TTL management, and pattern-based invalidation.

Critically, the caching layer must degrade gracefully. If Redis goes down, the API must continue working (just slower) rather than failing entirely. Cache is an optimization, not a dependency.

## User Story

**Como** developer usando o marketplace, **Quero** que listagens e validações respondam rapidamente, **Para que** minha experiência de busca e instalação de módulos seja fluida.

## Acceptance Criteria

- [ ] `@fastify/redis` plugin registered with ioredis client
- [ ] CacheService provides get, set, invalidate, invalidatePattern methods
- [ ] Module listings cached with 5-minute TTL
- [ ] Module listing cache invalidated on module create, update, or delete
- [ ] License validation results cached with 1-minute TTL
- [ ] License cache invalidated on revoke, expire, or update
- [ ] User session data cached tied to JWT expiry TTL
- [ ] Paddle product mapping cached with 1-hour TTL
- [ ] Cache key format: `kaven:{domain}:{identifier}` (e.g., `kaven:modules:list:1:20:newest`)
- [ ] CacheService.invalidatePattern() supports glob patterns (e.g., `kaven:modules:*`)
- [ ] Graceful degradation: if Redis is down, operations bypass cache and hit DB directly
- [ ] Redis connection errors are logged but do not throw
- [ ] Cache hit/miss is logged at debug level for observability
- [ ] Cache TTLs are configurable via environment variables
- [ ] Stale-while-revalidate pattern for module listings (serve stale, refresh in background)

## Technical Notes

### CacheService Implementation

```typescript
// src/services/cache.service.ts

import Redis from 'ioredis';

export interface CacheConfig {
  host: string;
  port: number;
  password?: string;
  keyPrefix: string;
  ttls: {
    moduleListing: number;    // seconds
    licenseValidation: number;
    paddleProducts: number;
    session: number;
  };
}

export class CacheService {
  private redis: Redis | null = null;
  private connected = false;

  constructor(private config: CacheConfig) {}

  async connect(): Promise<void> {
    try {
      this.redis = new Redis({
        host: this.config.host,
        port: this.config.port,
        password: this.config.password,
        keyPrefix: this.config.keyPrefix,
        lazyConnect: true,
        retryStrategy: (times) => {
          if (times > 3) {
            console.warn('Redis connection failed after 3 retries, running without cache');
            return null;
          }
          return Math.min(times * 500, 3000);
        },
      });

      this.redis.on('connect', () => {
        this.connected = true;
        console.log('Redis cache connected');
      });

      this.redis.on('error', (err) => {
        this.connected = false;
        console.warn('Redis cache error:', err.message);
      });

      this.redis.on('close', () => {
        this.connected = false;
      });

      await this.redis.connect();
    } catch (err) {
      console.warn('Failed to connect to Redis cache:', (err as Error).message);
      this.redis = null;
    }
  }

  async get<T>(key: string): Promise<T | null> {
    if (!this.isAvailable()) return null;

    try {
      const raw = await this.redis!.get(key);
      if (!raw) {
        console.debug(`Cache MISS: ${key}`);
        return null;
      }
      console.debug(`Cache HIT: ${key}`);
      return JSON.parse(raw) as T;
    } catch (err) {
      console.warn(`Cache get error for ${key}:`, (err as Error).message);
      return null;
    }
  }

  async set(key: string, value: unknown, ttlSeconds?: number): Promise<void> {
    if (!this.isAvailable()) return;

    try {
      const serialized = JSON.stringify(value);
      if (ttlSeconds) {
        await this.redis!.set(key, serialized, 'EX', ttlSeconds);
      } else {
        await this.redis!.set(key, serialized);
      }
    } catch (err) {
      console.warn(`Cache set error for ${key}:`, (err as Error).message);
    }
  }

  async invalidate(key: string): Promise<void> {
    if (!this.isAvailable()) return;

    try {
      await this.redis!.del(key);
      console.debug(`Cache invalidated: ${key}`);
    } catch (err) {
      console.warn(`Cache invalidate error for ${key}:`, (err as Error).message);
    }
  }

  async invalidatePattern(pattern: string): Promise<void> {
    if (!this.isAvailable()) return;

    try {
      const fullPattern = `${this.config.keyPrefix}${pattern}`;
      let cursor = '0';
      do {
        const [nextCursor, keys] = await this.redis!.scan(
          cursor,
          'MATCH',
          fullPattern,
          'COUNT',
          100
        );
        cursor = nextCursor;
        if (keys.length > 0) {
          // Remove prefix because ioredis keyPrefix auto-adds it
          const unprefixedKeys = keys.map((k) => k.replace(this.config.keyPrefix, ''));
          await this.redis!.del(...unprefixedKeys);
        }
      } while (cursor !== '0');

      console.debug(`Cache invalidated pattern: ${pattern}`);
    } catch (err) {
      console.warn(`Cache invalidatePattern error for ${pattern}:`, (err as Error).message);
    }
  }

  isAvailable(): boolean {
    return this.connected && this.redis !== null;
  }

  async disconnect(): Promise<void> {
    if (this.redis) {
      await this.redis.quit();
      this.redis = null;
      this.connected = false;
    }
  }
}
```

### Cache Key Constants

```typescript
// src/config/cache-keys.ts

export const CACHE_KEYS = {
  moduleList: (page: number, limit: number, sort: string) =>
    `modules:list:${page}:${limit}:${sort}`,

  moduleDetail: (slug: string) =>
    `modules:detail:${slug}`,

  licenseValidation: (key: string) =>
    `license:${key}:valid`,

  userSession: (userId: string) =>
    `session:${userId}`,

  paddleProducts: () =>
    `paddle:products`,
} as const;

export const CACHE_PATTERNS = {
  allModules: 'modules:*',
  allLicenses: 'license:*',
  specificLicense: (key: string) => `license:${key}:*`,
} as const;
```

### Usage in Module Service

```typescript
// src/services/module.service.ts (modifications)

export class ModuleService {
  constructor(
    private prisma: PrismaClient,
    private cache: CacheService
  ) {}

  async list(page: number, limit: number, sort: string): Promise<PaginatedModules> {
    const cacheKey = CACHE_KEYS.moduleList(page, limit, sort);

    // Try cache first
    const cached = await this.cache.get<PaginatedModules>(cacheKey);
    if (cached) return cached;

    // DB query
    const result = await this.queryModules(page, limit, sort);

    // Cache with 5-min TTL
    await this.cache.set(cacheKey, result, 300);

    return result;
  }

  async create(data: CreateModuleInput): Promise<Module> {
    const module = await this.prisma.module.create({ data });

    // Invalidate all module listings
    await this.cache.invalidatePattern(CACHE_PATTERNS.allModules);

    return module;
  }

  async update(slug: string, data: UpdateModuleInput): Promise<Module> {
    const module = await this.prisma.module.update({
      where: { slug },
      data,
    });

    // Invalidate specific module + all listings
    await this.cache.invalidate(CACHE_KEYS.moduleDetail(slug));
    await this.cache.invalidatePattern(CACHE_PATTERNS.allModules);

    return module;
  }
}
```

### Fastify Plugin Registration

```typescript
// src/plugins/redis.ts

import fp from 'fastify-plugin';

export default fp(async function redisPlugin(fastify: FastifyInstance) {
  const cacheService = new CacheService({
    host: process.env.REDIS_HOST || 'localhost',
    port: parseInt(process.env.REDIS_PORT || '6379'),
    password: process.env.REDIS_PASSWORD,
    keyPrefix: 'kaven:',
    ttls: {
      moduleListing: parseInt(process.env.CACHE_TTL_MODULES || '300'),
      licenseValidation: parseInt(process.env.CACHE_TTL_LICENSE || '60'),
      paddleProducts: parseInt(process.env.CACHE_TTL_PADDLE || '3600'),
      session: parseInt(process.env.CACHE_TTL_SESSION || '900'),
    },
  });

  await cacheService.connect();

  fastify.decorate('cache', cacheService);

  fastify.addHook('onClose', async () => {
    await cacheService.disconnect();
  });
});

declare module 'fastify' {
  interface FastifyInstance {
    cache: CacheService;
  }
}
```
