## Escopo

### 1. AuthZ helpers (`apps/api/src/modules/auth/authz/scope.ts`)

```typescript
// Novo arquivo
export async function canActOnTenant(
  user: { id: string; tenantId: string; role: string },
  tenantId: string,
): Promise<boolean> {
  if (user.role === 'SUPER_ADMIN') return true;
  return user.tenantId === tenantId;
}

export async function canActOnBillingAccount(
  user: { id: string; tenantId: string; role: string },
  billingAccountId: string,
  prisma: PrismaClient,
): Promise<boolean> {
  if (user.role === 'SUPER_ADMIN') return true;
  // Owner path (atual): user √© TENANT_ADMIN do tenant cuja BA === billingAccountId.
  const ba = await prisma.billingAccount.findFirst({
    where: { id: billingAccountId, deletedAt: null },
    select: { ownerUserId: true, id: true },
  });
  if (!ba) return false;
  if (ba.ownerUserId === user.id) return true;
  // F2.4.6 futuro: BillingAccountMember (checagem extra) ‚Äî deixar extension point.
  return false;
}
```
